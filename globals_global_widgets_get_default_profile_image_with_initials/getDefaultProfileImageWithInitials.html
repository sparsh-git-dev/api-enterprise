Widget getDefaultProfileImageWithInitials(
    {required String? name,
    double? textSize,
    FontWeight? fontWeight,
    Color? color}) {
  return GetBuilder<ThemeController>(
    builder: (themeLogic) {
      return Text(
        name != null && name != "" ? name.toString()[0].toUpperCase() : "",
        style: AppStyle().profilePictureInitialsStyles.copyWith(
              fontSize: textSize ?? AppStyle.fontSize20,
              color: color??themeLogic.getSelectedPrimaryThemeColor,
              fontWeight: fontWeight ?? FontWeight.w700,
            ),
      );
    }
  );
}

List<Container> skillsChipsWithRemoveButton(
    list,
    bool viewMore,
    int chipsInitialCount,
    isProfileSkillsList,
    getKeyWorkFromList,
    Function removeChipFromListCallback) {
  List<Container> skillsChips = [];
  if (list.length > chipsInitialCount && viewMore == false) {
    for (int i = 0; i < chipsInitialCount; i++) {
      skillsChips.add(Container(
        height: 31,
        decoration: BoxDecoration(
          borderRadius: BorderRadius.circular(7),
          color: AppColors.blue2,
        ),
        child: Chip(
          deleteIcon: const Icon(Icons.clear_rounded,
              color: AppColors.grey7, size: 16.0),
          onDeleted: () {
            removeChipFromListCallback(i);
          },
          label: Padding(
              padding: const EdgeInsets.only(bottom: 5),
              child: Text(
                  getKeyWorkFromList
                      ? list[i]!['keyword'].toString().toLowerCase()
                      : isProfileSkillsList
                          ? list[i].toString().toLowerCase()
                          : list[i].skills!.keyword.toString().toLowerCase(),
                  style: AppStyle().chipStyle)),
          backgroundColor: AppColors.grey3,
          shape: RoundedRectangleBorder(
            borderRadius: BorderRadius.circular(7),
          ),
        ),
      ));
    }
  } else {
    for (int i = 0; i < list.length; i++) {
      skillsChips.add(Container(
        height: 31,
        decoration: BoxDecoration(
          borderRadius: BorderRadius.circular(7),
          color: AppColors.blue2,
        ),
        child: Chip(
          deleteIcon: const Icon(Icons.clear_rounded,
              color: AppColors.grey7, size: 16.0),
          onDeleted: () {
            removeChipFromListCallback(i);
          },
          label: Padding(
              padding: const EdgeInsets.only(bottom: 5),
              child: Text(
                  getKeyWorkFromList
                      ? list[i]!['keyword'].toString().toLowerCase()
                      : isProfileSkillsList
                          ? list[i].toString().toLowerCase()
                          : list[i].skills!.keyword.toString().toLowerCase(),
                  style: AppStyle().chipStyle)),
          backgroundColor: AppColors.grey3,
          shape: RoundedRectangleBorder(
            borderRadius: BorderRadius.circular(7),
          ),
        ),
      ));
    }
  }
  return skillsChips;
}

PopupMenuItem<GlobalSearchMenuItem> buildMenuItem(GlobalSearchMenuItem item) {
  return PopupMenuItem<GlobalSearchMenuItem>(
    value: item,
    child: Text(
      item.text,
      style: AppStyle().regularStyle.copyWith(
          color: AppColors.black,
          fontSize: AppStyle.fontSize14,
          fontWeight: FontWeight.w400),
    ),
    textStyle: AppStyle().regularStyle.copyWith(
        color: AppColors.black,
        fontSize: AppStyle.fontSize12,
        fontWeight: FontWeight.w400),
  );
}

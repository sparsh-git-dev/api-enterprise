Container CandidateCardGlobalSearch(
    CandidateDataModel candidatesModel,
    int index,
    fromTalentPool,
    candidatesSelected,
    context,
    navigateToScreen,
    setState,
    shortlistCandidate,
    addToTalentPoolDialog,
    unselectSelectAllCheckbox) {
  return Container(
    child: GestureDetector(
      behavior: HitTestBehavior.opaque,
      onTap: () {
        navigateToScreen(candidatesModel);
      },
      child: Column(
        children: [
          Padding(
            padding: const EdgeInsets.only(
                top: 30.0, left: 5.40, right: 8.0, bottom: 0),
            child: Row(
              mainAxisAlignment: MainAxisAlignment.start,
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                if (fromTalentPool)
                  Transform.translate(
                    offset: const Offset(-2, 18),
                    child: Checkbox(
                      shape: RoundedRectangleBorder(
                          borderRadius: BorderRadius.circular(4)),
                      visualDensity: const VisualDensity(
                        horizontal: -4,
                        vertical: -3,
                      ),
                      // fillColor: MaterialStateProperty.resolveWith<Color>(
                      //     (Set<MaterialState> states) {
                      //   return AppColors.primaryPurple;
                      // }),
                      value: candidatesSelected.contains(candidatesModel.id),
                      onChanged: (bool? value) {
                        setState(() {
                          if (value!) {
                            candidatesSelected.add(candidatesModel.id);
                          } else {
                            candidatesSelected.remove(candidatesModel.id);
                          }
                          unselectSelectAllCheckbox(value);
                          print(candidatesSelected);
                        });
                      },
                    ),
                  ),
                const SizedBox(
                  width: 10,
                ),
                CircularProfileAvatar(
                  mediaFile: candidatesModel.mediaFile ?? "",
                  name: candidatesModel.firstName ?? "",
                  mediaFileSigned: candidatesModel.mediaFileSigned ?? "",
                  talentPoolStatus: candidatesModel.talentPool ?? "",
                  imageAvatarRadius: Get.width * 0.086,
                  videoAvatarRadius: Get.width * 0.186,
                ),
                const SizedBox(
                  width: 20,
                ),
                Column(
                  mainAxisAlignment: MainAxisAlignment.spaceEvenly,
                  crossAxisAlignment: CrossAxisAlignment.start,
                  mainAxisSize: MainAxisSize.min,
                  children: [
                    const SizedBox(
                      height: 4,
                    ),
                    _getCandidateName(candidatesModel, context),
                    getNewRatingDisplay(
                      candidateId: candidatesModel.id.toString(),
                      candidateRating: candidatesModel.candidateRating,
                      noOfRatings: candidatesModel.noOfRatings,
                      ratingSectionCallback: () {
                        navigateToScreen(candidatesModel);
                      },
                      styleVisibility: false,
                      textWidth: Get.width * 0.20,
                    ),
                    const SizedBox(height: 8),
                    candidatesModel.city != ''
                        ? _getHeroIconAndLabel(
                            HeroIcons.mapPin,
                            ConstantMethods().removeSpecialCharacter(
                                    candidatesModel.city!) +
                                ", " +
                                candidatesModel.country!)
                        : const SizedBox.shrink()
                  ],
                ),
                //const Spacer(),
                if (!fromTalentPool)
                  Expanded(
                    child: Transform.translate(
                      offset: const Offset(0, -15),
                      child: PopupMenuButton<GlobalSearchMenuItem>(
                        icon:
                            GetBuilder<ThemeController>(builder: (themeLogic) {
                          return Container(
                            padding: const EdgeInsets.symmetric(
                                vertical: 0, horizontal: 4),
                            decoration: BoxDecoration(
                                color:
                                    themeLogic.getSelectedSecondaryThemeColor,
                                borderRadius: BorderRadius.circular(4)),
                            child: Icon(
                              Icons.more_horiz,
                              color: themeLogic.getSelectedPrimaryThemeColor,
                            ),
                          );
                        }),
                        itemBuilder: (context) {
                          return [
                            ...globalSearchMenuItem.map(buildMenuItem).toList()
                          ];
                        },
                        onSelected: (item) async {
                          if (item.text == EnterpriseStrings.talent_pool_b005) {
                            Navigator.push(
                              context,
                              MaterialPageRoute(
                                builder: (context) => AddToTalentPoolScreen(
                                  totalCount: candidatesModel.totalCount,
                                  candidateId: candidatesModel.id.toString(),
                                  candidateName:
                                      "${candidatesModel.firstName} ${candidatesModel.lastName}",
                                ),
                              ),
                            );
                          }
                          if (item.text ==
                              EnterpriseStrings.viewCandidate_b007) {
                            List? result = await showShortlistDialog(
                              context,
                              candidatesModel.id.toString(),
                              candidateName: (candidatesModel.firstName ?? "") +
                                  " " +
                                  (candidatesModel.lastName ?? ""),
                              mediaFileSignedUrl:
                                  candidatesModel.portfolioUrl1.toString(),
                              availableDate:
                                  candidatesModel.availableDate.toString(),
                              talentPoolStatus:
                                  candidatesModel.talentPool.toString(),
                              location: candidatesModel.city.toString() +
                                  (candidatesModel.city!.isNotEmpty &&
                                          candidatesModel.country!.isNotEmpty
                                      ? ", "
                                      : '') +
                                  candidatesModel.country.toString(),
                            );
                            if (result!.isNotEmpty) {
                              // shortlist button is clicked
                              String firstName = candidatesModel.firstName
                                      .toString()
                                      .isNotEmpty
                                  ? candidatesModel.firstName.toString()
                                  : '';
                              String lastName =
                                  candidatesModel.lastName.toString().isNotEmpty
                                      ? candidatesModel.lastName.toString()
                                      : '';
                              InternetConnectivity()
                                  .checkConnectivity()
                                  .then((internet) {
                                internet
                                    ? shortlistCandidate(
                                        candidatesModel.id.toString(),
                                        firstName + " " + lastName,
                                        candidatesModel,
                                        index,
                                        result)
                                    : DialogsService().toastMessage(
                                        EnterpriseStrings.w001,
                                        AppColors.warningOpacity,
                                        context,
                                        8,
                                        AppColors.warning);
                              });
                            }
                          }
                        },
                      ),
                    ),
                  )
                //ThreeDotMenu(size: 3,)
              ],
            ),
          ),
          const SizedBox(
            height: 16,
          ),
          Padding(
            padding: EdgeInsets.only(left: fromTalentPool ? 50.0 : 16),
            child: Row(
              crossAxisAlignment: CrossAxisAlignment.center,
              mainAxisAlignment: MainAxisAlignment.start,
              children: [
                Container(
                    transform: Matrix4.translationValues(-5.0, 0.0, 0.0),
                    child: getStatusChip(
                        candidatesModel.availableDate!.isNotEmpty
                            ? 'MCO'
                            : candidatesModel.inCandidatePool == true
                                ? 'MCA'
                                : 'MBA',
                        fromMatchingCandidate: true,
                        context: context,
                        isAvailableStatus: true)),
                rehire(candidatesModel, context, false),
              ],
            ),
          ),
          if (getToolTipText(
            candidatesModel.availableDate!.isNotEmpty
                ? 'MCO'
                : candidatesModel.inCandidatePool == true
                    ? 'MCA'
                    : 'MBA',
          ).isNotEmpty) ...[
            Padding(
              padding: const EdgeInsets.only(
                left: 15.0,
                right: 0,
                top: 10,
              ),
              child: getTooltipStatus(
                  candidatesModel.availableDate!.isNotEmpty
                      ? 'MCO'
                      : candidatesModel.inCandidatePool == true
                          ? 'MCA'
                          : 'MBA',
                  candidatesModel.availableDate!),
            )
          ],
          (candidatesModel.availableDate!.isNotEmpty &&
                          candidatesModel.inCandidatePool!) ==
                      true &&
                  candidatesModel.availableToRehire!.isNotEmpty &&
                  candidatesModel.availableToRehire! == 'Y'
              ? Column(
                  children: [
                    const SizedBox(
                      height: 10,
                    ),
                    Container(
                      padding: const EdgeInsets.only(left: 15.0, right: 24),
                      alignment: Alignment.centerLeft,
                      child: Row(
                        mainAxisAlignment: MainAxisAlignment.start,
                        mainAxisSize: MainAxisSize.min,
                        children: [
                          rehire(candidatesModel, context, true),
                        ],
                      ),
                    ),
                  ],
                )
              : const SizedBox.shrink(),
          const SizedBox(
            height: 20,
          ),
          Padding(
            padding: EdgeInsets.only(left: fromTalentPool ? 50.0 : 16),
            child: Row(children: [
              Expanded(
                  flex: 1,
                  child: _getInfo('#Experience',
                      '${candidatesModel.overallYearsOfExperience?.toInt().toString()}${candidatesModel.overallYearsOfExperience! > 1 ? ' Years' : ' Year'}')),
              Expanded(
                  flex: 1,
                  child: _getInfo(
                      'Pay Rate',
                      candidatesModel.ratePerHour != null &&
                              candidatesModel.ratePerHour!.toInt() > 0
                          ? '${ConstantMethods().getCurrencyValue(candidatesModel.currency)} ${candidatesModel.ratePerHour?.toInt().toString()} / ${ConstantMethods().paymentTerm('HR')}'
                          : "-")),
              /*Expanded(flex:1,child: _getInfo('Availability',candidatesModel.availableDate != ''?DateConvertions().dateParse(
                    candidatesModel.availableDate
                        .toString(),
                  DateConvertions().outputDayFormat,
                    DateConvertions().dateFormats[2]!
                ):'-')),*/
            ]),
          ),
          const SizedBox(height: 30),
          const Divider(thickness: 1)
        ],
      ),
    ),
  );
}

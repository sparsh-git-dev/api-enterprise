Future<List<RejectionReasonJobsModel>> fetchFindAllJobsIfRejectionNotes(
    context,
    int page,
    String urlQueryParameters,
    String candidateId,
    bool isRejected) async {
  final String _urlParams = urlQueryParameters.replaceFirst("?", "&");
  var url = '';
  if (isRejected) {
    url = ApiUrlConstants.ENTERPRISE_REJECTION_NOTES_VIEW_JOBS +
        "/$candidateId?isRejectNote=true&searchType=prefix_match_phrase" +
        _urlParams;
  } else {
    url = ApiUrlConstants.ENTERPRISE_REJECTION_NOTES_VIEW_JOBS +
        "/$candidateId" +
        urlQueryParameters +
        "&searchType=prefix_match_phrase";
  }
  final response = await ApiServices().apiGetCallerWithOutLoader(url,
      context);
  Map<String, dynamic> responseMap = {"jobs": response};
  List<RejectionReasonJobsModel> _list = <RejectionReasonJobsModel>[];
  for (int i = 0; i < responseMap['jobs'].length; i++) {
    _list.add(RejectionReasonJobsModel.fromJson(responseMap['jobs'][i]));
  }
  return _list;
}

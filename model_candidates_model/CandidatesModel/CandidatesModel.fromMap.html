factory CandidatesModel.fromMap(Map<String, dynamic> json) {
  List<String> _fullLabelOfWorkTypes = <String>[];
  List<String> _fullLabelOfWorkplaces = <String>[];
  try {
    json['workType'] = json['workType']?.split(",") ?? [''];
  } catch (e) {
    debugPrint("onError: => workType: $e");
  }
  try {
    json['workPlace'] = json['workPlace']?.split(",") ?? [''];
  } catch (e) {
    debugPrint("onError: => workPlace: $e");
  }

  final List<Map<String, dynamic>> _workTypeList = [
    {"label": "Full Time", "short": "FT"},
    {"label": "Part Time", "short": "PT"},
    {"label": "Contract", "short": "FL"},
    {"label": "Internship", "short": "IN"},
    {"label": "Temporary", "short": "TP"},
    {"label": "Contingent", "short": "CG"},
    {"label": "Apprentice", "short": "AP"},
    {"label": "Other", "short": "OT"},
  ];
  final List<Map<String, dynamic>> _workPlaceList = [
    {"label": "Remote", "short": "R"},
    {"label": "Onsite", "short": "O"},
    {"label": "Hybrid", "short": "H"}
  ];

  ///Work Type
  for (String str in json['workType']) {
    for (int i = 0; i < _workTypeList.length; i++) {
      if (_workTypeList[i]['short'].toString().toLowerCase() ==
          str.toLowerCase()) {
        _fullLabelOfWorkTypes.add(_workTypeList[i]['label']);
      }
    }
  }

  ///Workplace
  for (String str in json['workPlace']) {
    for (int i = 0; i < _workPlaceList.length; i++) {
      if (_workPlaceList[i]['short'].toString().toLowerCase().trim() ==
          str.toLowerCase().trim()) {
        _fullLabelOfWorkplaces.add(_workPlaceList[i]['label']);
      }
    }
  }

  if (_fullLabelOfWorkTypes.isEmpty) {
    _fullLabelOfWorkTypes.add("-");
  }
  if (_fullLabelOfWorkplaces.isEmpty) {
    _fullLabelOfWorkplaces.add("-");
  }

  return CandidatesModel(
      mobileNumber: json["mobileNumber"] ?? '',
      hidePhoneEmail: json["hidePhoneEmail"] ?? '',
      emailAddress: json["emailAddress"] ?? '',
      firstName: json["firstName"] ?? '',
      lastName: json["lastName"] ?? '',
      knownAs: json["knownAs"] ?? '',
      nationality: json["nationality"] ?? '',
      address: json["address"] ?? '',
      mediaFile: json["mediaFile"] ?? '',
      mediaFileSigned: json["mediaFileSigned"] ?? '',
      backgroundMediaFile: json["backgroundMediaFile"] ?? '',
      backgroundMediaFileSigned: json["backgroundMediaFileSigned"] ?? '',
      availableForOpportunities: json["availableForOpportunities"] ?? '',
      brief: json["brief"] ?? '',
      portfolioUrl1: json["portfolioUrl1"] ?? '',
      portfolioUrl2: json["portfolioUrl2"] ?? '',
      portfolioUrl3: json["portfolioUrl3"] ?? '',
      portfolioUrl4: json["portfolioUrl4"] ?? '',
      mediaServicesUrl: json["mediaServicesUrl"] ?? '',
      profileTitle: json["profileTitle"] ?? '',
      neuroClassification: json["neuroClassification"] == "ND",
      skills: json["skills"] == null
          ? null
          : List<Skill>.from(json["skills"].map((x) => Skill.fromMap(x))),
      certificationList: json['certifications'] == null
          ? []
          : certificationListModelFromJson(
              jsonEncode(json['certifications'])),
      education: json["education"] == null
          ? null
          : List<Education>.from(
              json["education"].map((x) => Education.fromMap(x))),
      workHistory: json["workHistory"] == null
          ? null
          : List<WorkHistory>.from(
              json["workHistory"].map((x) => WorkHistory.fromMap(x))),
      resume: json["resume"] == null
          ? null
          : List<Resume>.from(json["resume"].map((x) => Resume.fromMap(x))),
      candidateLicenceDtos: json["candidateLicenceDtos"] == null
          ? null
          : List<CandidateLicenceDtos>.from(json["candidateLicenceDtos"]
              .map((x) => CandidateLicenceDtos.fromMap(x))),
      referencesDtos: json["referencesDtos"] == null
          ? null
          : List<ReferencesDtos>.from(
              json["referencesDtos"].map((x) => ReferencesDtos.fromMap(x))),
      ratePerHour: json["ratePerHour"] ?? '',
      overallYearsOfExperience: json["overallYearsOfExperience"] ?? '',
      profileCompletionPercentage: json["profileCompletionPercentage"] ?? 0.0,
      matchingScore: json["matchingScore"] ?? 0.0,
      noOfRatings: json["noOfRatings"],
      candidateRating: json["candidateRating"] ?? 0.0,
      availableDate: json["availableDate"] ?? "",
      inCandidatePool: json["inCandidatePool"] ?? false,
      workTypeFull: List<String>.from(_fullLabelOfWorkTypes.map((x) => x)),
      workPlace: json['workPlace'] ?? "-",
      city: json['city'] ?? "",
      state: json['state'] ?? "",
      country: json['country'] ?? "",
      workPlaceFull: List<String>.from(_fullLabelOfWorkplaces.map((x) => x)),
      talentPool: json["talentPool"] ?? '',
      zipCode: json['zipCode'] ?? "",
      currency: json["currency"] ?? '',
  );
}

void retainOnlyAppliedFilter(){
  ///Retain candidate type
  if(appliedFilters['candidateTypeIndex'] != null){
    updateCandidateTypes(appliedFilters['candidateTypeIndex']);
  } else {
    updateCandidateTypes(0);
  }
  ///Retain Location
  for(int i = 0; i < _selectedLocation.length; i++) {
    if(_selectedLocation[i]['isApplied'] == false) {
      removeLocation(i);
    }
  }
  ///Retain pay rate (Compensation)
  if(appliedFilters['payRate'] != null) {
    _rangeOfCompensation = appliedFilters['payRate'];
  } else {
    _rangeOfCompensation = const SfRangeValues(0, 100000);
  }

  ///Retain work type list
  if(appliedFilters['workTypeList'] != null) {
    for(int i = 0; i < workTypeList.length; i++){
      workTypeList[i]['isSelected'] = false;
    }
    for(int i = 0; i < appliedFilters['workTypeList'].length; i++) {
      for(int j = 0; j < workTypeList.length; j++) {
        if(workTypeList[j]['value'] == appliedFilters['workTypeList'][i]){
          workTypeList[j]['isSelected'] = true;
        }
      }
    }
  } else {
    for(int j = 0; j < workTypeList.length; j++) {
        workTypeList[j]['isSelected'] = true;
    }
  }
  ///Retain work place list
  if(appliedFilters['workPlaceList'] != null) {
    for(int i = 0; i < workPlaceList.length; i++){
      workPlaceList[i]['isSelected'] = false;
    }
    for(int i = 0; i < appliedFilters['workPlaceList'].length; i++) {
      for(int j = 0; j < workPlaceList.length; j++) {
        if(workPlaceList[j]['value'] == appliedFilters['workPlaceList'][i]){
          workPlaceList[j]['isSelected'] = true;
        }
      }
    }
  } else {
    for(int j = 0; j < workPlaceList.length; j++) {
      workPlaceList[j]['isSelected'] = true;
    }
  }
  ///Referred By
  if(appliedFilters['referredBy'] != null) {
    _isReferredCandidate = appliedFilters['referredBy'];
  } else {
    _isReferredCandidate = false;
  }
  if(appliedFilters['referrerFromSuggestion'] != null){
    _referredBySelectedFromSuggestion = appliedFilters['referrerFromSuggestion'];
    referredByTextEditingController.text = _referredBySelectedFromSuggestion!.fullName!.toString();
  } else {
    _referredBySelectedFromSuggestion = null;
    referredByTextEditingController.clear();
  }
  update();
}

Widget resumeData(
  CandidatesModel candidatesModel,
  BuildContext context,
  ScrollController? controller,
) {
  return Padding(
      padding: const EdgeInsets.only(top: 0, bottom: 0),
      child: candidatesModel.resume!.isNotEmpty
          ? ConstrainedBox(
              constraints: const BoxConstraints(
                maxHeight: 270.0,
              ),
              child: Scrollbar(
                thumbVisibility: candidatesModel.resume!.length > 3,
                controller: controller,
                child: ListView.builder(
                  controller: controller,
                  shrinkWrap: true,
                  itemCount: candidatesModel.resume!.length,
                  itemBuilder: (context, index) {
                    return Container(
                      margin: const EdgeInsets.only(bottom: 10),
                      padding: const EdgeInsets.fromLTRB(10, 17, 5, 17),
                      decoration: BoxDecoration(
                          borderRadius: BorderRadius.circular(8),
                          border: Border.all(
                              width: 1,
                              color: AppColors.profileScaffolds
                                  .withOpacity(0.2)),
                          color: AppColors.backgroundColor),
                      child: Row(
                        crossAxisAlignment: CrossAxisAlignment.start,
                        children: [
                          GestureDetector(
                              onTap: () async {
                                InternetConnectivity()
                                    .checkConnectivity()
                                    .then((internet) {
                                  internet
                                      ? launchWebUrl(candidatesModel
                                          .resume![index].url
                                          .toString())
                                      : DialogsService().toastMessage(
                                          EnterpriseStrings.w001,
                                          AppColors.warningOpacity,
                                          context,
                                          8,
                                          AppColors.warning);
                                });
                              },
                              child: SizedBox(
                                width: 30,
                                height: 30,
                                child: Image.asset(
                                  IconsDisplay().getIconForName(
                                      candidatesModel.resume![index].url
                                          .toString()
                                          .split('.')
                                          .last),
                                ),
                              )),
                          const SizedBox(width: 8),
                          GestureDetector(
                              behavior: HitTestBehavior.opaque,
                              onTap: () async {
                                String _fileName = candidatesModel
                                    .resume![index].urlSigned
                                    .toString()
                                    .split("?")
                                    .first
                                    .split("/")
                                    .last
                                    .replaceAll("%20", " ");
                                InternetConnectivity()
                                    .checkConnectivity()
                                    .then((internet) async {
                                  internet
                                      ? await DownloadFileFromUrl()
                                          .downloadFileURL(
                                              context,
                                              candidatesModel
                                                  .resume![index].urlSigned
                                                  .toString(),
                                              _fileName)
                                      : DialogsService().toastMessage(
                                          EnterpriseStrings.w001,
                                          AppColors.warningOpacity,
                                          context,
                                          8,
                                          AppColors.warning);
                                });
                              },
                              child: SizedBox(
                                width:
                                    MediaQuery.of(context).size.width / 1.3,
                                child: Row(
                                  crossAxisAlignment:
                                      CrossAxisAlignment.start,
                                  mainAxisAlignment:
                                      MainAxisAlignment.spaceBetween,
                                  children: [
                                    Column(
                                      crossAxisAlignment:
                                          CrossAxisAlignment.start,
                                      children: [
                                        Text(
                                          candidatesModel.resume![index].title
                                              .toString()
                                              .toCapitalCase(),
                                          overflow: TextOverflow.ellipsis,
                                          style: AppStyle()
                                              .regularStyle
                                              .copyWith(
                                                fontSize: AppStyle.fontSize16,
                                                fontWeight: FontWeight.w400,
                                              ),
                                        ),
                                        Padding(
                                          padding: const EdgeInsets.only(
                                              top: 12.0),
                                          child: Text(
                                            EnterpriseStrings
                                                    .viewCandidate_l033 +
                                                " " +
                                                (candidatesModel
                                                            .resume![index]
                                                            .lastUpdatedTimestamp !=
                                                        null
                                                    ? DateConvertions()
                                                        .dateParse(
                                                            candidatesModel
                                                                .resume![
                                                                    index]
                                                                .lastUpdatedTimestamp
                                                                .toString(),
                                                            DateConvertions()
                                                                .outputDayFormat,
                                                            DateConvertions()
                                                                .outputDayFormatPicker)
                                                    : '-'),
                                            overflow: TextOverflow.ellipsis,
                                            style: AppStyle()
                                                .regularStyle
                                                .copyWith(
                                                    fontSize:
                                                        AppStyle.fontSize14,
                                                    fontWeight:
                                                        FontWeight.w300,
                                                    color: AppColors
                                                        .subTitleColor),
                                          ),
                                        ),
                                      ],
                                    ),
                                    Container(
                                      margin: const EdgeInsets.only(top: 5),
                                      width: 25,
                                      height: 25,
                                      child: SvgPicture.asset(
                                          ImagePaths.ic_downloadSVG),
                                    ),
                                  ],
                                ),
                              )),
                        ],
                      ),
                    );
                  },
                ),
              ),
            )
          : const SizedBox.shrink());
}

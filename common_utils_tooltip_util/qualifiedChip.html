Widget qualifiedChip(String txt, bool isTablet) {
  return Container(
    height: isTablet ? 30 : 23,
    width: isTablet ? 140 : 90,
    padding: const EdgeInsets.symmetric(horizontal: 10),
    decoration: BoxDecoration(
        color: txt == "P" ? AppColors.oldLace : AppColors.lavenderMist,
        borderRadius: BorderRadius.circular(24)),
    child: Row(
      mainAxisAlignment: MainAxisAlignment.center,
      crossAxisAlignment: CrossAxisAlignment.center,
      children: [
        txt == "P"
            ? const HeroIcon(
                HeroIcons.lockClosed,
                style: HeroIconStyle.mini,
                color: AppColors.qualified,
                size: 10,
              )
            : const HeroIcon(
                HeroIcons.star,
                style: HeroIconStyle.mini,
                color: AppColors.earthBlue,
                size: 10,
              ),
        const SizedBox(width: 3),
        Text(
          statusTalentPoolChangesChip(txt),
          style: txt == "P"
              ? AppStyle().privateTextStyle
              : AppStyle().priorityTextStyle,
        )
      ],
    ),
  );
}

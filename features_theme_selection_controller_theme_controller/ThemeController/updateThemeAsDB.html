void updateThemeAsDB(dynamic enterpriseDetailsResponse) {
  // final response = await ApiServices().apiGetCallerWithOutLoader(ApiUrlConstants.ENTERPRISES, context);
  _enterpriseDetailsFromDB = enterpriseDetailsResponse;
  if(_enterpriseDetailsFromDB['primaryColor'] == null){
    _enterpriseDetailsFromDB['primaryColor'] = '228B22'; ///By default primary color
    _enterpriseDetailsFromDB['secondaryColor'] = 'EBFFEB'; ///By default secondary color
    enterpriseDetailsResponse['primaryColor'] = '228B22';
    enterpriseDetailsResponse['secondaryColor'] = 'EBFFEB';
  }
  if (enterpriseDetailsResponse['theme'] == null) {
    updateThemeIndex(0);
  } else if (enterpriseDetailsResponse['theme'] == 'Custom') {
    Color primaryColor = Color(int.parse(
        'FF' + enterpriseDetailsResponse['primaryColor'],
        radix: 16)); // Converting hex to color
    Color secondaryColor = Color(int.parse(
        'FF' + enterpriseDetailsResponse['secondaryColor'],
        radix: 16)); // Converting hex to color
    setCustomPrimaryThemeColor(primaryColor);
    setCustomSecondaryThemeColor(secondaryColor);
    setCustomThemeAsTrue();
  } else {
    // if(enterpriseDetailsResponse['theme'].toString().toLowerCase() == 'Forest Fern'.toLowerCase()){
    //   updateThemeIndex(0);
    // } else {
    bool isFoundInList = false;
      for (int i = 0; i < getListOfThemeColors.length; i++) {
        if (enterpriseDetailsResponse['theme'] == getListOfThemeColors[i].title) {
          updateThemeIndex(i);
          isFoundInList = true;
          break;
        }
      }
      if(!isFoundInList){
        Color primaryColor = Color(int.parse(
            'FF' + enterpriseDetailsResponse['primaryColor'],
            radix: 16)); // Converting hex to color
        Color secondaryColor = Color(int.parse(
            'FF' + enterpriseDetailsResponse['secondaryColor'],
            radix: 16)); // Converting hex to color
        setCustomPrimaryThemeColor(primaryColor);
        setCustomSecondaryThemeColor(secondaryColor);
        setCustomThemeAsTrue();
      }
    // }
  }
}

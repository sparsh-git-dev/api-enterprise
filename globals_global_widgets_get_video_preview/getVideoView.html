getVideoView(mediaUrl, context, double size, title) {
  return Center(
    child: Container(
      height: size,
      width: size,
      decoration: AppStyle().imageBorderStyle,
      child: Stack(
        children: [
          Padding(
            padding: AppStyle().imageBorderPadding,
            child: GetBuilder<ThemeController>(builder: (themeLogic) {
              return CircleAvatar(
                  backgroundColor: themeLogic.getSelectedSecondaryThemeColor,
                  child: getDefaultProfileImageWithInitials(name: null),
                  // backgroundImage:
                  //     const NetworkImage(AppUrl.DEFAULT_PROFILE_IMAGE_URL),
                  radius: size);
            }),
          ),
          GestureDetector(
            onTap: () {
              addProfileVideoDialog(context, mediaUrl, title);
              // launchWebUrl(mediaUrl);
            },

              child: SvgPicture.asset(
                ImagePaths.ic_playLogoSVG,
                alignment: Alignment.center,
                fit: BoxFit.fitHeight,
                height: 25,
                width: 25,
                color: AppColors.black,
              ),
            ),
          ),
        ],
      ),
    ),
  );
}

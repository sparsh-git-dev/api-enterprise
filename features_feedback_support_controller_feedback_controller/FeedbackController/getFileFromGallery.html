Future<String> getFileFromGallery(BuildContext context) async {
  // Request external storage permission
  if (await Permission.storage.request().isGranted) {
    // Permission granted, proceed with file picker logic
    FilePickerResult? result = await FilePicker.platform.pickFiles(
      type: FileType.image,
    );
    if (result != null) {
      File file = File(result.files.single.path.toString());
      if(file.path != null && file.path.isNotEmpty) {
        if(file.path.split('.').last.toUpperCase() == 'JPG' || file.path.split('.').last.toUpperCase() == 'JPEG'||file.path.split('.').last.toUpperCase() == 'PNG'){
          return file.path;
        } else {
          DialogsService().toastMessage(EnterpriseStrings.invalidImage, AppColors.errorOpacity, context, 6, AppColors.error);
        }
      }
    } else {

      return '';
    }
  } else {
    return '';
  }
  return '';
}

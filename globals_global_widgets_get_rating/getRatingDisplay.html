    Widget getRatingDisplay(
    {required candidateRating,
    required noOfRatings,
    required styleVisibility,
    required Function() ratingSectionCallback,
    required candidateId,
    bool fromPublicProfile = false,
    double starSizes = 20}) {
  return
      // noOfRatings != null
      true
          ? GestureDetector(
              behavior: HitTestBehavior.opaque,
              onTap: () {
                if (noOfRatings != null) {
                  if (fromPublicProfile) {
                    ratingSectionCallback();
                  } else {
                    if (noOfRatings > 0) {
                      Get.dialog(Material(
                          borderOnForeground: false,
                          type: MaterialType.transparency,
                          child: GetRatingDetails(candidateId: candidateId)));
                    }
                  }
                }
              },
              child: Row(
                children: [
                  if (!fromPublicProfile || candidateRating != 0)
                    getRatingBarWithRating(
                        starSizes,
                        candidateRating != null
                            ? double.parse(candidateRating!.toString())
                            : 0.0,
                        false),
                  // if (candidateRating != 0)
                  const SizedBox(width: 8),
                  if (candidateRating != 0)
                    GestureDetector(
                      behavior: HitTestBehavior.opaque,
                      onTap: () {
                        if (fromPublicProfile) {
                          ratingSectionCallback();
                        } else {
                          if (noOfRatings > 0) {
                            Get.dialog(Material(
                                borderOnForeground: false,
                                type: MaterialType.transparency,
                                child: GetRatingDetails(
                                    candidateId: candidateId)));
                          }
                        }
                      },
                      child: Padding(
                        padding: const EdgeInsets.only(top: 8),
                        child:
                            GetBuilder<ThemeController>(builder: (themeLogic) {
                          return Text(
                            noOfRatings != null
                                ? '(${noOfRatings!})'
                                // ' ${noOfRatings <= 1 ? EnterpriseStrings.rating_l024 : EnterpriseStrings.rating_l023})'
                                : "",
                            style: AppStyle().ratingsStyle.copyWith(
                                fontWeight: styleVisibility
                                    ? FontWeight.w600
                                    : FontWeight.w500,
                                color: themeLogic.getSelectedPrimaryThemeColor),
                          );
                        }),
                      ),
                    )
                ],
              ),
            )
          : const SizedBox.shrink();
}

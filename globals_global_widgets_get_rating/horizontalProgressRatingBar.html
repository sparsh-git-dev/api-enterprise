Widget horizontalProgressRatingBar(
    BuildContext context, String ratingNumber, double ratingPercent) {
  return Column(
    children: [
      Row(
        children: [
          Text(ratingNumber, style: AppStyle().regularStyle.copyWith()),
          const SizedBox(width: 8),
          Expanded(
            // width: MediaQuery.of(context).size.width / 1.6,
            child: ClipRRect(
              borderRadius: const BorderRadius.all(Radius.circular(4)),
              child: GetBuilder<ThemeController>(builder: (themeLogic) {
                return LinearProgressIndicator(
                  minHeight: 24,
                  value: ratingPercent / 100,
                  color: themeLogic.getSelectedPrimaryThemeColor,
                  backgroundColor: AppColors.textFormFieldFill,
                );
              }),
            ),
          ),
          const SizedBox(width: 8),
          SizedBox(
            width: 45,
            child: Text(
                ratingPercent.toInt().toString().replaceAll('.0', '') +
                    EnterpriseStrings.profile_l061,
                style: AppStyle().regularStyle.copyWith()),
          ),
        ],
      ),
      const SizedBox(height: 20),
    ],
  );
}

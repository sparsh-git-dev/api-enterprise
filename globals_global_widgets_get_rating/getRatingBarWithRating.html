Widget getRatingBarWithRating(double itemSize, double rating, bool isEditable) {
  return rating == 0
      ? Padding(
          padding: const EdgeInsets.only(top: 4.0),
          child: Text(
            EnterpriseStrings.rating_l019,
            style: AppStyle().regularStyle.copyWith(
                fontSize: AppStyle.fontSize13,
                fontWeight: FontWeight.w400,
                color: AppColors.subTitleColor),
          ),
        )
      : RatingBar(
          glow: false,
          itemSize: itemSize,
          initialRating: rating,
          direction: Axis.horizontal,
          allowHalfRating: false,
          itemCount: 5,
          ratingWidget: RatingWidget(
            full: SvgPicture.asset(
              ImagePaths.ic_activeRatingStarSVG,
              fit: BoxFit.none,
            ),
            empty: SvgPicture.asset(
              ImagePaths.ic_inactiveRatingStarSVG,
              fit: BoxFit.none,
            ),
            half: Image.asset(''),
          ),
          itemPadding: const EdgeInsets.only(
            top: 8,
          ),
          ignoreGestures: !isEditable,
          onRatingUpdate: (_) {},
        );
}

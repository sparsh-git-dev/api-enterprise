factory EnterprisesViewJobsDataModel.fromMap(Map<String, dynamic> json) {
  try {
    json['jobType'] = json['jobType']
        .replaceAll('[', '')
        .replaceAll(']', '')
        .replaceAll(' ', '')
        .split(",");
  } catch (e) {}
  return EnterprisesViewJobsDataModel(
    id: json["id"] ?? '',
    enterprisePublicDto: json["enterprisePublicDto"] == null
        ? null
        : EnterprisePublicDto.fromMap(json["enterprisePublicDto"]),
    jobDescriptionUrl: json["jobDescriptionUrl"] ?? '',
    jobDescriptionText: json["jobDescriptionText"] ?? '',
    jobTitle: json["jobTitle"] ?? '',
    budget: json["budget"] ?? 0,
    currency: json["currency"] ?? '',
    jobType: List<String>.from(json["jobType"].map((x) => x)),
    city: json["city"] ?? '',
    neuroClassification: json["neuroClassification"] == "ND",
    state: json["state"] ?? '',
    country: json["country"] ?? '',
    paymentTerm: json["paymentTerm"] ?? '',
    minimumEducation: json["minimumEducation"],
    nameOfHiringManager: json["nameOfHiringManager"] ?? '',
    numberOfOpenings: json["numberOfOpenings"] ?? '',
    benefits: json["benefits"] ?? '',
    aboutToExpire: json["aboutToExpire"] ?? '',
    jobManager: json["jobManager"] ?? '',
    jobSkills: json["jobSkills"] == null
        ? null
        : List<JobSkill>.from(
            json["jobSkills"].map((x) => JobSkill.fromMap(x))),
    status: json["status"] ?? '',
    numberOfApplicants: json["numberOfApplicants"] ?? '',
    validFrom:
        json["validFrom"] == null ? null : DateTime.parse(json["validFrom"]),
    validTill:
        json["validTill"] == null ? null : DateTime.parse(json["validTill"]),
    createdBy: json["createdBy"] ?? '',
    createdTimestamp: json["createdTimestamp"] == null
        ? null
        : DateTime.parse(json["createdTimestamp"]),
    lastUpdatedBy: json["lastUpdatedBy"] ?? '',
    lastUpdatedTimestamp: json["lastUpdatedTimestamp"] == null
        ? null
        : DateTime.parse(json["lastUpdatedTimestamp"]),
    refreshedOn: json["refreshedOn"] == null
        ? null
        : DateTime.parse(json["refreshedOn"]),
    minimumBudgetPerPaymentTerm: json["minimumBudgetPerPaymentTerm"] ?? '',
    maximumBudgetPerPaymentTerm: json["maximumBudgetPerPaymentTerm"] ?? '',
    limitForApplications: json["limitForApplications"] ?? '',
    limitForShortlists: json["limitForShortlists"] ?? '',
    overallExperienceMinimum: json["overallExperienceMinimum"] ?? '',
    overallExperienceMaximum: json["overallExperienceMaximum"] ?? '',
    budgetMinimum: json["budgetMinimum"] ?? '',
    checkListId: json["checkListId"] ?? -1,
    budgetMaximum: json["budgetMaximum"] ?? '',
    workFrom: json["workFrom"] ?? '',
    jobEducationalQualificationsDtoList:
        json["jobEducationalQualificationsDtoList"] == null
            ? []
            : List<JobEducationalQualificationsDtoList>.from(
                json["jobEducationalQualificationsDtoList"]!.map(
                  (x) => JobEducationalQualificationsDtoList.fromMap(x),
                ),
              ),
    //neuroClassifications: json["neuroClassification"] ?? ''
  );
}

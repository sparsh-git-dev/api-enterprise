Future<void> selectExcelFile(BuildContext context) async {
  FilePickerResult? _result;
  if(_selectedExcel.isEmpty){
    _result = await FilePicker.platform.pickFiles(
        type: FileType.custom,
        allowedExtensions: ['xls', 'xlsx', 'xlsm'],
        allowMultiple: false
    );
  } else {
    DialogsService().toastMessage(EnterpriseStrings.invalidNoOfExcelUpload, AppColors.errorOpacity, context, 8, AppColors.warning,alignTextCenter: true);
  }

  if(_result != null) {
    if (_selectedExcel.isEmpty) {
      _selectedExcel = _result.paths.map((path) => File(path!)).toList();
      _selectedExcel.retainWhere((element) {
        final String _extension = element.path.split('.').last;
        if (_extension == 'xlsx' || _extension == 'xlsm' || _extension == 'xls') {
          return true;
        } else {
          DialogsService().toastMessage(EnterpriseStrings.invalidFileUpload, AppColors.errorOpacity, context, 8, AppColors.warning);
          return false;
        }
      });
    } else {
      _selectedExcel.addAll(_result.paths.map((path) => File(path!)).toList());
      _selectedExcel.retainWhere((element) {
        final String _extension = element.path.split('.').last;
        if (_extension == 'xlsx' || _extension == 'xlsm' || _extension == 'xls') {
          return true;
        } else {
          DialogsService().toastMessage(EnterpriseStrings.invalidFileUpload, AppColors.errorOpacity, context, 8, AppColors.warning);
          return false;
        }
      });
    }
  } else {}
  update();
}

Future<List<NotesDatum>?> fetchNotes(context, String urlQueryParameters,
    String candidateId, int page, bool isRejection) async {
  if (isRejection) {
    final String _urlParams = urlQueryParameters.replaceFirst("?", "&");
    final response = await ApiServices().apiGetCallerWithOutLoader(
        ApiUrlConstants.ENTERPRISE_REJECTION_NOTES_VIEW_JOBS + '/$candidateId' + "?isRejectNote=true" + _urlParams,
        context);
    Map<String, dynamic> responseMap = {"data": response};
    NotesModel enterprisesViewJobsModel = NotesModel.fromMap(responseMap);
    return enterprisesViewJobsModel.data;
  } else {
    final response = await ApiServices().apiGetCallerWithOutLoader(
        ApiUrlConstants.ENTERPRISE_NOTES +
            '/$candidateId' +
            urlQueryParameters,
        context);
    Map<String, dynamic> responseMap = {"data": response};
    NotesModel enterprisesViewJobsModel = NotesModel.fromMap(responseMap);
    return enterprisesViewJobsModel.data;
  }
}

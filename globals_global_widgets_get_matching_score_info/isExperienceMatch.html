String isExperienceMatch(
    String preferredCandidateExperience, double candidateExperience) {
  String iconPath = '';
  double minExperience = 0.0;
  double maxExperience = 100;
  if (preferredCandidateExperience != '-') {
    minExperience = double.parse(preferredCandidateExperience.split('to')[0]);
    maxExperience = double.parse(preferredCandidateExperience.split('to').last);

    if (minExperience == maxExperience) {
      if (minExperience == candidateExperience) {
        iconPath = ImagePaths.ic_greenCheckLogoSVG;
      }
    } else if (maxExperience > minExperience) {
      if (candidateExperience >= minExperience ||
          candidateExperience >= maxExperience) {
        iconPath = ImagePaths.ic_greenCheckLogoSVG;
      }
    } else {
      iconPath = ImagePaths.ic_redCrossLogoSVG;
    }
  }
  if (iconPath == '') {
    iconPath = ImagePaths.ic_redCrossLogoSVG;
  }

  return iconPath;
}

Future<void> fetchTalentPoolCandidates(talentPoolId, BuildContext context,
    {append = false, showLoader = true, queryParams = ''}) async {
  // setIsLoading(true);
  if (!append) setInitialLoad(true);
  String url = ApiUrlConstants.TALENT_POOL_CREATE +
      "/" +
      talentPoolId.toString() +
      "/candidates/" +
      (candidates.length / 10).ceil().toString() +
      '/10' +
      (queryParams != '' ? queryParams : '');
  final GlobalKey<State> _keyLoader = GlobalKey<State>();
  final usersResponse;
  if (showLoader) {
    usersResponse =
        await ApiServices().apiGetCallerWithLoader(url, context, _keyLoader);
  } else {
    usersResponse =
        await ApiServices().apiGetCallerWithoutLoader(url, context);
  }
  var response =
      await ApiServices().apiGetCallerWithoutLoaderFullResponse(url, context);
  totalCount = json.decode(response.body)['totalCount'].toString();
  if (usersResponse == []) {
    setPaginationFlag(true);
  }
  setCandidatesList(myCandidatesResponseModelFromMap(usersResponse),
      append: append);
  setIsLoading(false);
  setInitialLoad(false);
}

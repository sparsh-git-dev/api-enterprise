Future<void> fetchMoreCandidates(BuildContext context) async {
  String url = ApiUrlConstants.MY_CANDIDATES_GET;
  List<String> _skillsList = <String>[];
  List<String> _selectedCity = <String>[];
  List<String> _selectedState = <String>[];
  List<String> _selectedCountry = <String>[];
  List<String> _selectedEducationList = <String>[];
  List<String> _selectedTalentPoolList = <String>[];
  final String _enterpriseId =
      await UserPreferences().getPreferenceValues(UserPreferences.prefId);
  for (int i = 0; i < getSelectedSkills.length; i++) {
    _skillsList.add(getSelectedSkills[i]['id'].toString());
  }
  for (int i = 0; i < getSelectedLocation.length; i++) {
    _selectedCity.add(getSelectedLocation[i]['city']!);
    _selectedState.add(getSelectedLocation[i]['state']!);
    _selectedCountry.add(getSelectedLocation[i]['country']!);
  }
  for (final element in _listEducation) {
    if (element['isSelected']) {
      _selectedEducationList.add(element['short']);
    }
  }
  for (final element in _listTalentPools) {
    if (element['isSelected']) {
      _selectedTalentPoolList.add(element['short']);
    }
  }
  final GlobalKey<State> _keyLoader = GlobalKey<State>();
  final data = {
    "enterpriseId": _enterpriseId,
    "skills": _skillsList,
    "company": getSelectedCompanies,
    "city": _selectedCity,
    "state": _selectedState,
    "country": _selectedCountry,
    "education": _selectedEducationList,
    "talentPool": _selectedTalentPoolList,
    "profileTitle": getSelectedJobTitles,
    "minYearsOfExperience": (getSfRangeValuesForExperience.start.round() == 0)
        ? "0"
        : getSfRangeValuesForExperience.start.round().toString(),
    "maxYearsOfExperience": (getSfRangeValuesForExperience.end.round() == 30)
        ? "30+"
        : getSfRangeValuesForExperience.end.round().toString(),
    "minPayRate": (getSfRangeValuesForCompensation.start.round() == 0)
        ? "0"
        : getSfRangeValuesForCompensation.start.round().toString(),
    "maxPayRate": (getSfRangeValuesForCompensation.end.round() == 100000)
        ? "100000+"
        : getSfRangeValuesForCompensation.end.round().toString(),
    "pageFrom": (candidates.length / 10).ceil().toString(),
    "pageSize": "10"
  };
  final usersResponse =
      await ApiServices().apiPutCallerFromServicesWithLoader(
    url,
    data,
    context,
    _keyLoader,
    true,
    true,
  );
  var newPosts = myCandidatesResponseModelFromMap(usersResponse);
  if (newPosts.isEmpty) {
    setHasReachedMax(true);
    setIsLoading(false);
  } else {
    setCandidatesList(newPosts, append: true);
    setIsLoading(false);
  }
}

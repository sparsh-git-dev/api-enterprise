Future<void> scroll(index, context) async {
  tabController.animateTo(index);
  if (isSuperAdmin) {
    switch (index) {
      case 0:
        {
          await scrollController
              .animateTo(
            scrollController.position.minScrollExtent,
            curve: Curves.easeOut,
            duration: const Duration(milliseconds: 1),
          )
              .then((value) {
            updateIndex(index);
          });
        }
        break;
      case 1:
        {
          await scrollController
              .animateTo(
            (Get.height / 1.8),
            curve: Curves.easeOut,
            duration: const Duration(milliseconds: 1),
          )
              .then((value) {
            updateIndex(index);
          });
        }
        break;
      case 2:
        {
          if (tabIndex == 1) {
            await scrollController
                .animateTo(
              scrollController.position.maxScrollExtent -
                  (MediaQuery.of(context).size.height / 1.8),
              curve: Curves.easeOut,
              duration: const Duration(milliseconds: 1),
            )
                .then((value) {
              updateIndex(index);
            });
          } else if (tabIndex == 0) {
            await scrollController
                .animateTo(
              scrollController.position.maxScrollExtent + 40,
              // (MediaQuery.of(context).size.height/3.4),
              curve: Curves.easeOut,
              duration: const Duration(milliseconds: 1),
            )
                .then((value) {
              updateIndex(index);
            });
          }
        }
        break;
      case 3:
        {
          await scrollController
              .animateTo(
            scrollController.position.maxScrollExtent,
            curve: Curves.easeOut,
            duration: const Duration(milliseconds: 1),
          )
              .then((value) {
            updateIndex(index);
          });
        }
      // break;
    }
  } else {
    switch (index) {
      case 0:
        {
          await scrollController.animateTo(
            scrollController.position.minScrollExtent,
            curve: Curves.easeOut,
            duration: const Duration(milliseconds: 1),
          );
        }
        break;
      case 1:
        {
          // await scrollController.animateTo(
          //   scrollController.position.maxScrollExtent,
          //   curve: Curves.easeOut,
          //   duration: const Duration(milliseconds: 1),
          // );
          await scrollController
              .animateTo(
            (Get.height / 1.8),
            curve: Curves.easeOut,
            duration: const Duration(milliseconds: 1),
          )
              .then((value) {
            updateIndex(index);
          });
        }
        break;
      case 2:
        {
          await scrollController
              .animateTo(
            scrollController.position.maxScrollExtent,
            curve: Curves.easeOut,
            duration: const Duration(milliseconds: 1),
          )
              .then((value) {
            updateIndex(index);
          });
        }
        break;
    }
  }
}

candidateRatingDialog(BuildContext context, String candidateName, jobTitle,
    jobId, candidateId, onCompleteCallback,
    {showPriorityCandidateSection = false}) {
  bool proceedSubmit = false;
  double overallRating = 0;
  double deliveryQualityRating = 0;
  double scheduleRating = 0;
  double stickingToCommitmentRating = 0;
  double valueForMoneyRating = 0;
  TextEditingController reviewController = TextEditingController();
  showDialog(
      context: context,
      barrierDismissible: false,
      builder: (context) {
        bool isRequestedToGoPublic = false;
        return StatefulBuilder(
          builder: (context, setState) => SingleChildScrollView(
            reverse: true,
            child: Dialog(
              insetPadding: const EdgeInsets.only(
                  left: 10, right: 10, top: 24, bottom: 24),
              child: Padding(
                padding: const EdgeInsets.only(
                    left: 24, right: 10, top: 24, bottom: 24),
                child: Column(
                  children: [
                    Row(
                      mainAxisAlignment: MainAxisAlignment.spaceBetween,
                      crossAxisAlignment: CrossAxisAlignment.start,
                      children: [
                        Flexible(
                          child: Text(
                            EnterpriseStrings.rating_m001 +
                                candidateName +
                                EnterpriseStrings.rating_m002 +
                                jobTitle.toString(),
                            maxLines: 3,
                            style: AppStyle().boldStyle.copyWith(
                                fontSize: AppStyle.fontSize18,
                                fontWeight: FontWeight.w700),
                          ),
                        ),
                        GestureDetector(
                            onTap: () {
                              Navigator.of(context).pop();
                            },
                            child: const Icon(
                              Icons.clear_rounded,
                              color: AppColors.subTitleColor,
                            ))
                      ],
                    ),
                    const SizedBox(height: 24),
                    SizedBox(
                      height: MediaQuery.of(context).size.width / 1,
                      child: ListView(
                        children: [
                          const TextLabel(
                              label: EnterpriseStrings.rating_l001,
                              isRequired: true),
                          const SizedBox(height: 5),
                          Align(
                              alignment: Alignment.centerLeft,
                              child: getRating(24, (value) {
                                setState(() {
                                  overallRating = value;
                                  if (overallRating > 0) {
                                    proceedSubmit = true;
                                  } else {
                                    proceedSubmit = false;
                                  }
                                });
                              })),
                          const SizedBox(height: 12),
                          const TextLabel(
                              label: EnterpriseStrings.rating_l003,
                              isRequired: false),
                          const SizedBox(height: 8),
                          Align(
                              alignment: Alignment.centerLeft,
                              child: getRating(24, (value) {
                                setState(() {
                                  deliveryQualityRating = value;
                                });
                              })),
                          const SizedBox(height: 16),
                          const TextLabel(
                              label: EnterpriseStrings.rating_l004,
                              isRequired: false),
                          const SizedBox(height: 8),
                          Align(
                              alignment: Alignment.centerLeft,
                              child: getRating(24, (value) {
                                setState(() {
                                  scheduleRating = value;
                                });
                              })),
                          const SizedBox(height: 16),
                          const TextLabel(
                              label: EnterpriseStrings.rating_l005,
                              isRequired: false),
                          const SizedBox(height: 8),
                          Align(
                              alignment: Alignment.centerLeft,
                              child: getRating(24, (value) {
                                setState(() {
                                  stickingToCommitmentRating = value;
                                });
                              })),
                          const SizedBox(height: 16),
                          const TextLabel(
                              label: EnterpriseStrings.rating_l006,
                              isRequired: false),
                          const SizedBox(height: 8),
                          Align(
                              alignment: Alignment.centerLeft,
                              child: getRating(24, (value) {
                                setState(() {
                                  valueForMoneyRating = value;
                                });
                              })),
                          const SizedBox(height: 25),
                          const TextLabel(
                              label: EnterpriseStrings.rating_l007,
                              isRequired: false),
                          const SizedBox(height: 16),
                          CustomTextFormField(
                            maxLines: 5,
                            controller: reviewController,
                            keyboardType: TextInputType.text,
                            textCapitalization: TextCapitalization.sentences,
                            validator: (text) {
                              return Validations().singleValidation(
                                  text.toString().trim(), 256);
                            },
                            autovalidateMode:
                                AutovalidateMode.onUserInteraction,
                            onChanged: (value) {
                              setState(() {
                                if (value[0].endsWith(' ')) {
                                  reviewController.selection =
                                      const TextSelection(
                                          baseOffset: 0, extentOffset: 0);
                                  reviewController.text = value.trim();
                                } else {
                                  Validations().singleValidation(
                                      value.toString().trim(), 256);
                                }
                              });
                            },
                            hintText: EnterpriseStrings.rating_h001,
                            filled: true,
                          ),
                        ],
                      ),
                    ),
                    const SizedBox(height: 16),
                    if (showPriorityCandidateSection)
                      Column(
                        children: [
                          Transform.translate(
                            offset: const Offset(-15, 0),
                            child: Row(
                              crossAxisAlignment: CrossAxisAlignment.center,
                              children: [
                                StatefulBuilder(builder: (context, setStater) {
                                  return Transform.scale(
                                    scale: 0.6,
                                    child: CupertinoSwitch(
                                        trackColor: AppColors.grey,
                                        activeColor: AppColors.primaryPurple,
                                        value: isRequestedToGoPublic,
                                        onChanged: (bool value) {
                                          setState(() {
                                            isRequestedToGoPublic = value;
                                          });
                                        }),
                                  );
                                }),
                                Text(
                                  EnterpriseStrings.talent_pool_l023,
                                  style: AppStyle().textStyleLabel.copyWith(
                                        color: AppColors.black,
                                        fontSize: AppStyle.fontSize13,
                                      ),
                                ),
                              ],
                            ),
                          ),
                          const SizedBox(height: 8),
                          Row(
                            crossAxisAlignment: CrossAxisAlignment.start,
                            children: [
                              Padding(
                                padding: const EdgeInsets.only(top: 2.0),
                                child: SizedBox(
                                  height: 15,
                                  width: 15,
                                  child: SvgPicture.asset(
                                    ImagePaths.ic_infoSVG,
                                    fit: BoxFit.fitHeight,
                                    color: AppColors.subTitleColor,
                                  ),
                                ),
                              ),
                              const SizedBox(width: 6),
                              SizedBox(
                                width: MediaQuery.of(context).size.width - 120,
                                child: Text(
                                  EnterpriseStrings.talent_pool_l025,
                                  maxLines: 4,
                                  overflow: TextOverflow.ellipsis,
                                  style: AppStyle().regularStyle.copyWith(
                                        fontSize: AppStyle.fontSize12,
                                        fontWeight: FontWeight.w400,
                                        color: AppColors.subTitleColor,
                                      ),
                                ),
                              ),
                            ],
                          ),
                        ],
                      ),
                    const SizedBox(height: 18),
                    Row(
                      mainAxisAlignment: MainAxisAlignment.spaceBetween,
                      children: [
                        GestureDetector(
                          onTap: () {
                            // onCancelCallback();
                            InternetConnectivity()
                                .checkConnectivity()
                                .then((internet) {
                              if (internet) {
                                if (isRequestedToGoPublic) {
                                  final TalentPoolController
                                      talentPoolController =
                                      Get.put(TalentPoolController());
                                  final data = [candidateId];
                                  talentPoolController
                                      .putRequestsTalentPool(data, "S", context)
                                      .then((value) {
                                    MatchingCandidatesCommonMethods()
                                        .rateYourExperienceDialog(
                                            jobId,
                                            candidateId,
                                            0,
                                            0,
                                            0,
                                            0,
                                            0,
                                            "",
                                            context, () {
                                      onCompleteCallback();
                                    });
                                  });
                                } else {
                                  MatchingCandidatesCommonMethods()
                                      .rateYourExperienceDialog(
                                          jobId,
                                          candidateId,
                                          0,
                                          0,
                                          0,
                                          0,
                                          0,
                                          "",
                                          context, () {
                                    onCompleteCallback();
                                  });
                                }
                              } else {
                                DialogsService().toastMessage(
                                    EnterpriseStrings.w001,
                                    AppColors.warningOpacity,
                                    context,
                                    8,
                                    AppColors.warning);
                              }
                            });
                          },
                          child: Container(
                            constraints: const BoxConstraints(maxWidth: 100),
                            child: Text(
                              EnterpriseStrings.b047,
                              style: AppStyle().regularStyleLink.copyWith(
                                    fontSize: AppStyle.fontSize14,
                                    fontWeight: FontWeight.w400,
                                  ),
                            ),
                          ),
                        ),
                        Container(
                          padding: const EdgeInsets.only(right: 5),
                          height: 55,
                          width: MediaQuery.of(context).size.width / 2,
                          child: CustomElevatedButton(
                            isEnable: proceedSubmit &&
                                reviewController.text.length <= 256,
                            onPressed: () {
                              if (proceedSubmit &&
                                  reviewController.text.length <= 256) {
                                InternetConnectivity()
                                    .checkConnectivity()
                                    .then((internet) {
                                  if (internet) {
                                    if (isRequestedToGoPublic) {
                                      final TalentPoolController
                                          talentPoolController =
                                          Get.put(TalentPoolController());
                                      final data = [candidateId];
                                      talentPoolController
                                          .putRequestsTalentPool(
                                              data, "S", context)
                                          .then((value) {
                                        MatchingCandidatesCommonMethods()
                                            .rateYourExperienceDialog(
                                                jobId,
                                                candidateId,
                                                overallRating,
                                                deliveryQualityRating,
                                                scheduleRating,
                                                stickingToCommitmentRating,
                                                valueForMoneyRating,
                                                reviewController.text
                                                    .toString(),
                                                context, () {
                                          onCompleteCallback();
                                        });
                                      });
                                    } else {
                                      MatchingCandidatesCommonMethods()
                                          .rateYourExperienceDialog(
                                              jobId,
                                              candidateId,
                                              overallRating,
                                              deliveryQualityRating,
                                              scheduleRating,
                                              stickingToCommitmentRating,
                                              valueForMoneyRating,
                                              reviewController.text.toString(),
                                              context, () {
                                        onCompleteCallback();
                                      });
                                    }
                                  } else {
                                    DialogsService().toastMessage(
                                        EnterpriseStrings.w001,
                                        AppColors.warningOpacity,
                                        context,
                                        8,
                                        AppColors.warning);
                                  }
                                });
                              }
                            },
                            labelText: EnterpriseStrings.b009,
                            isWidthInfinity: false,
                          ),
                        ),
                      ],
                    ),
                  ],
                ),
              ),
            ),
          ),
        );
      });
}

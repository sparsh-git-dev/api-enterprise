downloadProfileRequestToGetURLs(
    profileCheckboxSelected,
    selectedChecklists,
    resumeSelectedValues,
    directory,
    candidateId,
    firstName,
    doNotPop,
    context,
    mode,
    jobId) async {
  bool resumeSelected = false;
  for (int i = 0; i < resumeSelectedValues.length; i++) {
    if (resumeSelectedValues[i].value == true) {
      resumeSelected = true;
      break;
    }
  }
  Map<String, dynamic> paramsStatusCapture = {};
  if (mode != 'view-only') {
    paramsStatusCapture = {
      "jobId": jobId,
      "isProfileDownloaded": profileCheckboxSelected ? true : null,
      "isResumeDownloaded": resumeSelected ? true : null,
      "candidateId": candidateId,
    };
  } else {
    paramsStatusCapture = {
      "candidateId": candidateId,
      "isProfileDownloaded": profileCheckboxSelected ? true : null,
      "isResumeDownloaded": resumeSelected ? true : null,
    };
  }
  MatchingCandidatesRepo()
      .statusCaptureByProfileView(context, paramsStatusCapture);
  DownloadProfileResponseModel? downloadProfileResponseModel;
  final GlobalKey<State> _keyLoader = GlobalKey<State>();
  Map<String, dynamic> params = {
    "profile": profileCheckboxSelected,
    "checkList": selectedChecklists
  };
  await MatchingCandidatesRepo()
      .postDownloadProfileRequestToGetURL(
    params,
    candidateId.toString(),
    context,
    _keyLoader,
  )
      .then((data) async {
    final fetchedData = {"data": data};
    // setState(() {
    downloadProfileResponseModel =
        DownloadProfileResponseModel.fromMap(fetchedData);
    // });
    downloadFilesFromURLs(
      downloadProfileResponseModel!,
      resumeSelectedValues,
      directory,
      firstName,
      context,
    );
    if (!doNotPop) {
      Navigator.of(context).pop();
    }
  });
}

void shortlistOrRejectDialogCandidate(
    jobId, candidateId, context, Function callback,
    {bool isReject = false}) {
  TextEditingController applicationRemarksController = TextEditingController();
  showDialog(
      context: context,
      barrierDismissible: false,
      builder: (context) {
        return Dialog(
          insetPadding: const EdgeInsets.symmetric(horizontal: 16),
          backgroundColor: Colors.white,
          shape: RoundedRectangleBorder(
            borderRadius: BorderRadius.circular(8),
          ),
          elevation: 4,
          child: Padding(
            padding: const EdgeInsets.symmetric(vertical: 16, horizontal: 16),
            child: Column(
              mainAxisSize: MainAxisSize.min,
              children: <Widget>[
                Row(
                  mainAxisAlignment: MainAxisAlignment.spaceBetween,
                  children: [
                    Align(
                      alignment: Alignment.centerLeft,
                      child: Text(
                        isReject
                            ? EnterpriseStrings.viewCandidate_l024
                            : EnterpriseStrings.viewCandidate_l010,
                        style: AppStyle().boldStyle.copyWith(
                              fontSize: AppStyle.fontSize16,
                              fontWeight: FontWeight.w700,
                            ),
                      ),
                    ),
                    GestureDetector(
                      onTap: () {
                        applicationRemarksController.clear();
                        Navigator.of(context, rootNavigator: true).pop();
                      },
                      child: const Align(
                        alignment: Alignment.centerRight,
                        child: Icon(
                          Icons.clear_rounded,
                          color: AppColors.subTitleColor,
                        ),
                      ),
                    ),
                  ],
                ),
                const SizedBox(height: 24),
                CustomTextFormField(
                  maxLines: 5,
                  minLines: 5,
                  controller: applicationRemarksController,
                  keyboardType: TextInputType.text,
                  textCapitalization: TextCapitalization.sentences,
                  hintText: (isReject)
                      ? EnterpriseStrings.viewCandidate_h007
                      : EnterpriseStrings.viewCandidate_h002,
                  hintMaxLines: 8,
                ),
                const SizedBox(height: 24),
                SizedBox(
                  width: MediaQuery.of(context).size.width,
                  height: 42,
                  child: CustomElevatedButton(
                    labelText: isReject
                        ? EnterpriseStrings.viewCandidate_b003
                        : EnterpriseStrings.viewCandidate_b007,
                    isWidthInfinity: true,
                    onPressed: () {
                      if (isReject) {
                        InternetConnectivity()
                            .checkConnectivity()
                            .then((internet) {
                          internet
                              ? MatchingCandidatesCommonMethods()
                                  .rejectCandidateJob(
                                      applicationRemarksController.text
                                          .toString()
                                          .trim(),
                                      jobId,
                                      candidateId,
                                      context, () {
                                  callback();
                                })
                              : DialogsService().toastMessage(
                                  EnterpriseStrings.w001,
                                  AppColors.warningOpacity,
                                  context,
                                  8,
                                  AppColors.warning);
                        });
                      } else {
                        InternetConnectivity()
                            .checkConnectivity()
                            .then((internet) {
                          internet
                              ? MatchingCandidatesCommonMethods()
                                  .shortlistCandidate(
                                      applicationRemarksController.text
                                          .toString(),
                                      jobId.toString(),
                                      candidateId,
                                      context, () {
                                  FocusScope.of(context)
                                      .requestFocus(FocusNode());
                                  DialogsService().toastMessage(
                                      EnterpriseStrings.mEnterprise002,
                                      AppColors.successOpacity,
                                      context,
                                      6,
                                      AppColors.success);
                                  Navigator.pop(context);
                                  Navigator.of(context, rootNavigator: true)
                                      .pop();
                                })
                              : DialogsService().toastMessage(
                                  EnterpriseStrings.w001,
                                  AppColors.warningOpacity,
                                  context,
                                  8,
                                  AppColors.warning);
                        });
                      }
                    },
                  ),
                ),
              ],
            ),
          ),
        );
      });
}

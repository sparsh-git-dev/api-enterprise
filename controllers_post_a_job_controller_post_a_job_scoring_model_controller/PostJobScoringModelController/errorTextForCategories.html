String? errorTextForCategories(TextEditingController _controller, int index) {
  /// using index we get know which text editing controller is that

  switch(index) {
    case 0:
      final text = _controller.value.text;
      if (text.isEmpty) {
        return 'Skill weightage is required';
      }
      if (int.parse(text)<20) {
        return "Skill weightage can not be less than 20";
      }else if(int.parse(text)>100) {
        return "Skill weightage can not be more than 100";
      } else {
        return null;
      }
    case 1:
      final text = _controller.value.text;
      if (text.isEmpty) {
        return "Industry weightage is required";
      }
      else if(int.parse(text)>100) {
        return "Industry weightage can not be more than 100";
      } else {
        return null;
      }
    case 2:
      final text = _controller.value.text;
      if (text.isEmpty) {
        return "Overall experience weightage is required";
      }
      else if(int.parse(text)>100) {
        return "Overall experience can not be more than 100";
      } else {
        return null;
      }
    case 3:
      final text = _controller.value.text;
      if (text.isEmpty) {
        return "Profile title weightage is required";
      }
      else if(int.parse(text)>100) {
        return "Profile title can not be more than 100";
      } else {
        return null;
      }
    case 4:
      final text = _controller.value.text;
      if (text.isEmpty) {
        return "Education weightage is required";
      }
      else if(int.parse(text)>100) {
        return "Education weightage can not be more than 100";
      } else {
        return null;
      }
    case 5:
      final text = _controller.value.text;

        return "Certification weightage is required";
      }
      else if(int.parse(text)>100) {
        return "Certification can not be more than 100";
      } else {
        return null;
      }
      ///After 6 ageing sections
    default:
      return null;
  }
}

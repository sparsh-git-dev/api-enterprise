factory MyCandidatesResponseModel.fromMap(Map<String, dynamic> json) =>
    MyCandidatesResponseModel(
        candidateName: json["candidateName"] ?? '',
        profileTitle: json["profileTitle"] ?? '',
        emailAddress: json["emailAddress"] ?? '',
        hidePhoneEmail: json["hidePhoneEmail"] ?? '',
        country: json["country"] ?? '',
        state: json["state"] ?? '',
        city: json["city"] ?? '',
        availableDate: json["availableDate"] ?? '',
        availableToRehire: json["availableToRehire"] ?? '',
        profileImageUrl: json["profileImageUrl"] ?? '',
        profileImageUrlSigned: json["profileImageUrlSigned"] ?? '',
        totalYearsOfExperience: json["totalYearsOfExperience"] ?? '',
        candidateSkills:
            List<String>.from(json["candidateSkills"].map((x) => x)),
        noOfProjectsAssociated: json["noOfProjectsAssociated"] ?? '',
        noOfProjectsCompleted: json["noOfProjectsCompleted"] ?? '',
        noOfProjectsWorking: json["noOfProjectsWorking"] ?? '',
        noOfRatings: json["noOfRatings"],
        candidateRating: json["candidateRating"] ?? 0,
        projectsAssociatedList: List<ProjectList>.from(
            json["projectsAssociatedList"]
                .map((x) => ProjectList.fromMap(x))),
        projectCompletedList: List<ProjectList>.from(
            json["projectCompletedList"].map((x) => ProjectList.fromMap(x))),
        projectWorkingList: List<ProjectList>.from(
            json["projectWorkingList"].map((x) => ProjectList.fromMap(x))),
        candidateId: json["candidateId"] ?? '',
        talentPool: json["talentPool"] ?? '',
        source: json["source"],
        matchedJobs: json["matchedJobs"],
        inCandidatePool: json["inCandidatePool"] ?? false);

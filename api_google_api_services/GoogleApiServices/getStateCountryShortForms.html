getStateCountryShortForms(String query, context) async {
  final GlobalKey<State> _keyLoader = GlobalKey<State>();
  var totalAddress = [];
  if (query.isNotEmpty) {
    await ApiServices()
        .apiGetCallerWithLoader(
            ApiUrlConstants.GOOGLE_CITIES_API + query + "?type=D",
            context,
            _keyLoader)
        .then((response) {
      var adr = response[0]['addressComponents'];
      for (int i = 0; i < adr.length; i++) {
        if (adr[i]['types'][0] == "ADMINISTRATIVE_AREA_LEVEL_1") {
          totalAddress.add(adr[i]['shortName']);
        }
        if (adr[i]['types'][0] == "COUNTRY") {
          totalAddress.add(adr[i]['shortName']);
        }
      }
    }).catchError((onError) => print("onError:$onError"));
  }
  return totalAddress;
}

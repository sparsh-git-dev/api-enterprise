Future<dynamic> apiCallerFromServices(String url, dynamic data,
    BuildContext context, GlobalKey<State> _keyLoader,
    {required String method,
    bool isPublicAPI = false,
    bool isXFormEncoded = false,
    bool isLoader = true,
    var header,
    bool useReturnResponse = true}) async {
  if (isLoader) {
    DialogsService().showLoadingDialog(context, _keyLoader);
  }
  var responseJson;
  try {
    String fullUrl = url;
    print("----URL->>$fullUrl");
    Map<String, String> headers = {};
    if (header == null) {
      if (isPublicAPI) {
        headers = {
          'Content-Type': 'application/json',
          'Accept': 'application/json',
        };
      } else {
        headers = await createAuthorizationHeader(context,
            isXFormEncoded: isXFormEncoded);
      }
    } else {
      headers = header;
    }
    dynamic body = '';
    if (data != null && !isXFormEncoded) {
      body = json.encode(data);
    }

    http.Response response;
    switch (method) {
      case 'GET':
        response = await http.get(Uri.parse(fullUrl), headers: headers);
        break;
      case 'POST':
        log(fullUrl + "----" + headers.toString() + '====' + body);
        response = await http.post(Uri.parse(fullUrl),
            headers: headers, body: body.isNotEmpty ? body : data);
        log(fullUrl + "----" + response.toString());
        break;
      case 'PUT':
        response = await http.put(Uri.parse(fullUrl),
            headers: headers, body: body.isNotEmpty ? body : data);
        break;
      default:
        throw Exception('Unsupported HTTP method: $method');
    }

    if (isLoader) {
      Navigator.of(_keyLoader.currentContext!, rootNavigator: true).pop();
    }
    log("-------->>>///" + response.body.toString());
    // log("---????" + json.decode(response.body).toString());
    responseJson = useReturnResponse
        ? returnResponseForCredentialing(response, context)
        : response.body;
    // log("-------->>>" + responseJson.toString());
    return responseJson;
  } catch (e) {
    throw Exception('Error==========$e');
  }
}

dynamic returnResponseForCredentialing(http.Response response, context,
    {hideToastMessage = false}) {
  var responseJson = json.decode(response.body);
  print('status code===${response.statusCode}');
  switch (response.statusCode) {
    case 200:
      return responseJson['data'];
    case 400:
      throw response.body;
    case 401:
      throw response.body;
    case 404:
      DialogsService().toastMessage(
          'Unable to connect to server. Please refresh and try again',
          AppColors.errorOpacity,
          context,
          10,
          AppColors.error);
      throw response.body;
    case 403:
      signOut(context);
      return responseJson;
    case 500:
      if (!hideToastMessage) {
        if (responseJson['status'] != null) {
          serverErrorMessage(responseJson, context);
        } else {
          DialogsService().toastMessage(
              'Ooops, something went wrong!!! This problem has been reported to the System Administrator.',
              AppColors.errorOpacity,
              context,
              10,
              AppColors.error);
        }
      }
      throw response.body;
    default:
      DialogsService().toastMessage(
          'Ooops, something went wrong!!! This problem has been reported to the System Administrator.',
          AppColors.errorOpacity,
          context,
          10,
          AppColors.error);
      throw response.body;
  }
}

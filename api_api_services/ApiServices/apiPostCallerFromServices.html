Future apiPostCallerFromServices(url, data, context, _keyLoader,
    {isPublicAPI = false, isXFormEncoded = false, isLoader = true,isJobWIzard = false}) async {
  if (isLoader) {
    DialogsService().showLoadingDialog(context, _keyLoader);
  }
  var responseJson;
  try {
    print("----URL->>${(isPublicAPI ? BASE_URL_PUBLIC : BASE_URL) + url}");
    var headers = await createAuthorizationHeader(context,
        isXFormEncoded: isXFormEncoded);
    var body = '';
    if (!isXFormEncoded) {
      body = json.encode(data);
    }
    var response = await http.post(
      Uri.parse((isPublicAPI ? BASE_URL_PUBLIC : BASE_URL) + url),
      headers: headers,
      body: body.isNotEmpty ? body : data,
    );
    if (isLoader) {
      Navigator.of(_keyLoader.currentContext, rootNavigator: true).pop();
    }
    if(isJobWIzard == true && response.statusCode == 502){
      return 502;
    }
    responseJson = returnResponse(response, context);
    return responseJson;
  } catch (e) {
    throw Exception('Error==========$e');
  }
}

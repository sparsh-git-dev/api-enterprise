List<Widget> viewMoreText(
    list,
    bool viewMore,
    int chipsInitialCount,
    ismoreTextList,
    String labelText,
    Function onTapCallback,
    bool showText,
    bool isFullView,
    {bool enable = true}) {
  if (enable == false) {
    return [Container()];
  }
  List<Widget> moreText = [];
  if (list.length > chipsInitialCount && viewMore == false) {
    for (int i = 0; i < chipsInitialCount; i++) {
      moreText.add(Align(
        alignment: Alignment.topLeft,
        child:
        RichText(
          maxLines: 2,
            text: TextSpan(
              children: [
                TextSpan(
                  text: '• ',
                  style:  AppStyle().regularStyle.copyWith(
                      color: AppColors.dot,
                      fontSize: AppStyle.fontSize14,
                      fontWeight: FontWeight.w400,
                      height: 1.5),
                ),
                TextSpan(
                  text: '${(list[i].toString())}',
                  style:   AppStyle().regularStyle.copyWith(
                      color: AppColors.grey9,
                      overflow: TextOverflow.ellipsis,
                      fontSize: AppStyle.fontSize14,
                      fontWeight: FontWeight.w400,
                      height: 1.5)
                ),
              ],
            ))
      ));
    }
  } else {
    for (int i = 0; i < list.length; i++) {
      moreText.add(
        Align(
          alignment: Alignment.topLeft,
          child: Text("• ${list[i].toString() + (list[i] == list.last ? '' : ',')}",
              textAlign: TextAlign.left,
              style: AppStyle().regularStyle.copyWith(
                  color: AppColors.grey9,
                  fontSize: AppStyle.fontSize14,
                  fontWeight: FontWeight.w400,
                  height: 1.5)),
        ),
      );
    }
  }
  if (showText == true) {
    moreText.add(
      labelText != ''
          ? GetBuilder<ThemeController>(
          builder: (themeLogic) {
            return  GestureDetector(
        onTap: () {
              onTapCallback();
        },
        child: Transform.translate(
              offset: Offset(0, isFullView ? 2 : 0),
              child: Text(
                labelText,
                textAlign: TextAlign.end,
                style: AppStyle().viewMoreLessStyle.copyWith(
                  fontSize: AppStyle.fontSize14,
                  fontWeight: FontWeight.w600,
                  color: themeLogic.getSelectedPrimaryThemeColor
                ),
              ),
        ),
      );
            }
          )
          : const SizedBox(),
    );
  }
  return moreText;
}

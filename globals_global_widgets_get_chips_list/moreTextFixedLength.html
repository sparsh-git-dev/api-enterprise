List<Widget> moreTextFixedLength(
    list,
    bool viewMore,
    int chipsInitialCount,
    ismoreTextList,
    String labelText,
    Function onTapCallback,
    bool showText,
    bool isFullView,
    {bool enable = true}) {
  if (enable == false) {
    return [Container()];
  }
  List<Widget> moreText = [];
  if (list.length > chipsInitialCount && viewMore == false) {
    for (int i = 0; i < chipsInitialCount; i++) {
      moreText.add(Text(list[i].toString(),
          style: AppStyle().regularStyle.copyWith(
              //height: 12,
              fontSize: AppStyle.fontSize12,
              fontWeight: FontWeight.w400,
              height: 1.5)));
    }
  } else {
    for (int i = 0; i < list.length; i++) {
      moreText.add(
        Text(list[i].toString() + (list[i] == list.last ? '' : ','),
            style: AppStyle().regularStyle.copyWith(
                fontSize: AppStyle.fontSize12,
                fontWeight: FontWeight.w400,
                height: 1.5)),
      );
    }
  }
  if (showText == true) {
    moreText.add(
      labelText != ''
          ? GestureDetector(
              onTap: () {
                onTapCallback();
              },
              child: Transform.translate(
                offset: Offset(0, isFullView ? 3 : 1),
                child: Text(
                  labelText,
                  textAlign: TextAlign.end,
                  style: AppStyle().viewMoreLessStyle.copyWith(
                        fontSize: AppStyle.fontSize12,
                        fontWeight: FontWeight.w600,
                      ),
                ),
              ),
            )
          : const SizedBox(),
    );
  }
  return moreText;
}

List<Container> skillsChipsWithFixedLength(list, bool viewMore,
    int chipsInitialCount, isProfileSkillsList, BuildContext context) {
  List<Container> skillsChips = [];
  if (list.length > chipsInitialCount && viewMore == false) {
    for (int i = 0; i < chipsInitialCount; i++) {
      skillsChips.add(Container(
        height: Responsive.isTablet(context) ? 50 : 32,
        decoration: BoxDecoration(
          borderRadius: BorderRadius.circular(17),
          color: AppColors.textFormFieldFill,
        ),
        child: Chip(
          label: Padding(
              padding: const EdgeInsets.only(bottom: 5),
              child: Text(
                  isProfileSkillsList
                      ? list[i].toString().toLowerCase()
                      : list[i].skills!.keyword.toString().toLowerCase(),
                  style: AppStyle().chipStyle)),
          backgroundColor: AppColors.textFormFieldFill,
          shape: RoundedRectangleBorder(
            borderRadius: BorderRadius.circular(17),
          ),
        ),
      ));
    }
  } else {
    for (int i = 0; i < list.length; i++) {
      skillsChips.add(Container(
        height: Responsive.isTablet(context) ? 50 : 32,
        decoration: BoxDecoration(
          borderRadius: BorderRadius.circular(17),
          color: AppColors.textFormFieldFill,
        ),
        child: Chip(
          label: Padding(
              padding: const EdgeInsets.only(bottom: 5),
              child: Text(
                  isProfileSkillsList
                      ? list[i].toString().toLowerCase()
                      : list[i].skills!.keyword.toString().toLowerCase(),
                  style: AppStyle().chipStyle)),
          backgroundColor: AppColors.textFormFieldFill,
          shape: RoundedRectangleBorder(
            borderRadius: BorderRadius.circular(17),
          ),
        ),
      ));
    }
  }
  return skillsChips;
}

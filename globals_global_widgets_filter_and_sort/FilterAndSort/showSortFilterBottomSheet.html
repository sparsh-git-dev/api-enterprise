    Future<void> showSortFilterBottomSheet(
  BuildContext context,
  int selectedFilterFrom,
  bool isSortSelected,
  bool isFilterSelected,
  int selectedTabIndex,
  Function onSortApplied,
  Function updateSortValues,
  int? sortValue, {
  String? id,
  void Function()? callBack,
  String? modeSort,
}) async {
  final FilterAndSortController filterAndSortController =
      Get.put(FilterAndSortController());

  filterAndSortController.controllerInit(selectedFilterFrom);

  await showModalBottomSheet(
      context: context,
      isScrollControlled: true,
      builder: (context) {
        return StatefulBuilder(
            builder: (BuildContext context, StateSetter setBottomState) {
          return GetBuilder<FilterAndSortController>(builder: (logic) {
            return SingleChildScrollView(
              child: Padding(
                padding: EdgeInsets.only(
                    left: 16,
                    right: 16,
                    top: 14,
                    bottom: MediaQuery.of(context).viewInsets.bottom),
                child: Column(
                  children: [
                    Row(
                      mainAxisAlignment: MainAxisAlignment.spaceBetween,
                      children: [
                        SizedBox(
                          height: 38,
                          width: Responsive.isTablet(context) ? 95 : 85,
                          child: ElevatedButton(
                            style: ElevatedButton.styleFrom(
                                elevation: 0,
                                primary: isSortSelected
                                    ? null
                                    : AppColors.textFormFieldFill,
                                shape: RoundedRectangleBorder(
                                  borderRadius: BorderRadius.circular(4),
                                )),
                            onPressed: () {
                              setBottomState(() {
                                isSortSelected = true;
                                isFilterSelected = false;
                              });
                            },
                            child: Text(
                              EnterpriseStrings.viewCandidate_b008,
                              style: AppStyle().regularStyle.copyWith(
                                    fontSize: AppStyle.fontSize16,
                                    fontWeight: FontWeight.w400,
                                    color: isSortSelected
                                        ? AppColors.white
                                        : AppColors.textFormFieldHint
                                            .withOpacity(0.6),
                                  ),
                            ),
                          ),
                        ),
                        SizedBox(
                          height: 38,
                          width: Responsive.isTablet(context) ? 105 : 85,
                          child: ElevatedButton(
                            style: ElevatedButton.styleFrom(
                                elevation: 0,
                                primary: isFilterSelected
                                    ? null
                                    : AppColors.textFormFieldFill,
                                shape: RoundedRectangleBorder(
                                  borderRadius: BorderRadius.circular(4),
                                )),
                            onPressed: () {
                              setBottomState(() {
                                isSortSelected = false;
                                isFilterSelected = true;
                              });
                            },
                            child: Text(
                              EnterpriseStrings.viewCandidate_b009,
                              style: AppStyle().regularStyle.copyWith(
                                    fontSize: AppStyle.fontSize16,
                                    fontWeight: FontWeight.w400,
                                    color: isFilterSelected
                                        ? AppColors.white
                                        : AppColors.textFormFieldHint
                                            .withOpacity(0.6),
                                  ),
                            ),
                          ),
                        ),
                        const Spacer(),
                        GestureDetector(
                            onTap: () {
                              setBottomState(() {
                                logic.setGreySpaceSelected(false);
                              });
                              logic.clearButtonOnPress();
                              Navigator.of(context, rootNavigator: true)
                                  .pop();
                            },
                            child: const Icon(Icons.clear_rounded,
                                color: AppColors.subTitleColor)),
                      ],
                    ),
                    const SizedBox(height: 16),
                    isSortSelected == true && selectedTabIndex == 0
                        ? sortBody(onSortApplied, updateSortValues,
                            sortValue!, logic, selectedFilterFrom)

                        ///This was implemented by UMANG JAIN BUT COMMENTING AND DOING NEW ONE BECAUSE OF MATCHING SCORE FILTER REQUIREMENT, IF need to revert just uncomment => MatchingCandidatesBottomSheetFilter()
                        // : MatchingCandidatesBottomSheetFilter()
                        : MatchingWeightageFilter(
                            jobId: id!,
                            modeOfSort: modeSort!,
                            changeGreySpace: () => setBottomState(() {
                                  logic.setGreySpaceSelected(false);
                                }),
                            resetAllCallback: callBack!)
                  ],
                ),
              ),
            );
          });
        });
      }).then((value) {
    if (filterAndSortController.isGreySpaceSelected) {
      filterAndSortController.clearButtonOnPress();
    }
  });
}

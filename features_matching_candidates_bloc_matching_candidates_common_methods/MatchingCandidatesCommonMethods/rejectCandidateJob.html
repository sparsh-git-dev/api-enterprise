Future rejectCandidateJob(
    String reason, jobId, candidateId, context, onCompleteCallback) async {
  final params = {
    'jobId': jobId.toString(),
    'candidateId': candidateId.toString(),
    'rejectionReason': reason.trim()
  };
  final GlobalKey<State> _keyLoader = GlobalKey<State>();
  await MatchingCandidatesRepo()
      .putCandidatesReject(params, context, _keyLoader)
      .then((data) {
    if (data['id'] != null) {
      onCompleteCallback();
    }
  });
}

  String convertedDate = DateConvertions().dateParse(
      offerDate,
      DateConvertions().outputDayFormatPicker,
      DateConvertions().outputDayFormat);
  String convertedStartDate = '';
  String convertedEndDate = '';
  if (startDate != '' && endDate != '') {
    convertedStartDate = DateConvertions().dateParse(
        startDate,
        DateConvertions().outputDayFormatPicker,
        DateConvertions().outputDayFormat);
    convertedEndDate = DateConvertions().dateParse(
        endDate,
        DateConvertions().outputDayFormatPicker,
        DateConvertions().outputDayFormat);
  }
  String date = DateFormat('hh:mm:ss').format(DateTime.now());
  Map<String, dynamic> params = {};
  params = {
    'jobId': jobId.toString(),
    'candidateId': candidateId.toString(),
    'offerExpiryDate': convertedDate + "T" + date,
  };

  if (startDate != '' && endDate != '') {
    params["jobStartDate"] = convertedStartDate + "T" + date;
    params["jobCompletionDate"] = convertedEndDate + "T" + date;
  }
  if (payRate != '') {
    params["payRate"] = "$payRate";
  }
  if (workflowUniqueIdentifier != '') {
    params["workflowUniqueIdentifier"] = "$workflowUniqueIdentifier";
  }
  print("params");
  print(params);
  final GlobalKey<State> _keyLoader = GlobalKey<State>();
  // if (isAttachment) {
  await MatchingCandidatesRepo()
      .putAttachmentsSendRequest(
          params, context, filePath, isAttachment, _keyLoader)
      .then((value) {
    print("value");
    print(value);
    onCompleteCallback(value);
  });
}

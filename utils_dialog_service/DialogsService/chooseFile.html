chooseFile(context, Function callback) async {
  late File _educationFile;
  FilePickerResult? result = await FilePicker.platform.pickFiles(
    type: FileType.custom,
    allowedExtensions: ['pdf', 'doc', 'docx', 'zip'],
  );
  if (result != null) {
    PlatformFile file = result.files.first;
    _educationFile = File(result.files.toString());
    String? contentType =
        lookupMimeType(fileUtil.basename(file.path.toString()));
    Future.delayed(const Duration(milliseconds: 500), () async {
      callback(contentType, file, result);
    });
  } else {
    // User canceled the picker
  }
}

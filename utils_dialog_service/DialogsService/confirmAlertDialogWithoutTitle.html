Future<bool> confirmAlertDialogWithoutTitle(BuildContext context,
    String title, String message, String cancelStr, String okStr) async {
  bool action = false;
  await showDialog(
    context: context,
    barrierDismissible: false,
    builder: (context) {
      return Dialog(
        backgroundColor: Colors.white,
        shape: RoundedRectangleBorder(
          borderRadius: BorderRadius.circular(4),
        ),
        elevation: 4,
        child: Padding(
          padding: const EdgeInsets.symmetric(
            horizontal: 15,
            vertical: 15,
          ),
          child: IntrinsicWidth(
            child: IntrinsicHeight(
              child: Column(
                children: <Widget>[
                  title != ''
                      ? const SizedBox(
                          height: 20,
                        )
                      : const SizedBox(
                          height: 0,
                        ),
                  title != ''
                      ? Align(
                          alignment: Alignment.centerLeft,
                          child: Text(
                            title,
                            style: AppStyle().dialogStyle.copyWith(
                                fontWeight: FontWeight.w700,
                                fontSize: AppStyle.fontSize16),
                          ),
                        )
                      : Container(),
                  const SizedBox(
                    height: 20,
                  ),
                  Text(
                    message,
                    style: AppStyle()
                        .dialogStyle
                        .copyWith(fontSize: AppStyle.fontSize16),
                  ),
                  const SizedBox(
                    height: 20,
                  ),
                  Row(
                    mainAxisAlignment: MainAxisAlignment.center,
                    crossAxisAlignment: CrossAxisAlignment.center,
                    children: <Widget>[
                      ButtonTheme(
                          minWidth: 100.0,
                          height: 35.0,
                          child: MaterialButton(
                              highlightColor: Colors.transparent,
                              elevation: 1,
                              shape: RoundedRectangleBorder(
                                  side: const BorderSide(
                                      color: AppColors.grey, width: 1),
                                  borderRadius: BorderRadius.circular(4)),
                              color: AppColors.white,
                              onPressed: () {
                                action = false;
                                Navigator.of(context, rootNavigator: true)
                                    .pop();
                              },
                              child: Padding(
                                padding: const EdgeInsets.only(bottom: 3),
                                child: Text(
                                  cancelStr,
                                  style: AppStyle().dialogStyle.copyWith(
                                        fontSize: AppStyle.fontSize16,
                                        color: AppColors.footerBlack,
                                      ),
                                  textAlign: TextAlign.center,
                                ),
                              ))),
                      const Padding(padding: EdgeInsets.only(left: 10)),
                      ButtonTheme(
                          minWidth: 100.0,
                          height: 35.0,
                          child: GetBuilder<ThemeController>(
                              builder: (themeLogic) {
                            return MaterialButton(
                                highlightColor: themeLogic
                                    .getSelectedPrimaryThemeColor, //AppColors.primaryHoverButtonColor,
                                elevation: 1,
                                shape: RoundedRectangleBorder(
                                  borderRadius: BorderRadius.circular(4),
                                ),
                                color: themeLogic
                                    .getSelectedPrimaryThemeColor, //AppColors.primaryButtonColor,
                                onPressed: () {
                                  action = true;
                                  Navigator.of(context, rootNavigator: true)
                                      .pop();
                                },
                                child: Padding(
                                  padding: const EdgeInsets.only(bottom: 3),
                                  child: Text(
                                    okStr,
                                    style: AppStyle().dialogStyle.copyWith(
                                        fontSize: AppStyle.fontSize16,
                                        color: AppColors.white),
                                    textAlign: TextAlign.center,
                                  ),
                                ));
                          })),
                    ],
                  ),
                ],
              ),
            ),
          ),
        ),
      );
    },
  );

}

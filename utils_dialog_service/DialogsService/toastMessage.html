void toastMessage(String message, Color color, BuildContext context,
    int duration, Color textColor,
    {alignTextCenter = false}) {
  Widget _widget = Container(
    width: MediaQuery.of(context).size.width - 32,
    decoration: BoxDecoration(
      color: color,
      borderRadius: BorderRadius.circular(6.0),
    ),
    child: Padding(
      padding:
          const EdgeInsets.only(left: 10.0, right: 10, top: 15, bottom: 15),
      child: Row(crossAxisAlignment: CrossAxisAlignment.start, children: [
        Padding(
          padding: const EdgeInsets.only(left: 8),
          child: SvgPicture.asset(
            getAlertIcon(color),
            fit: BoxFit.cover,
          ),
        ),
        Flexible(
          child: Padding(
            padding: EdgeInsets.only(
                left: 5.0, right: 5.0, top: alignTextCenter ? 3 : 0),
            child: Text(message,
                textAlign: TextAlign.start,
                style: AppStyle().textStyleLabel.copyWith(
                      // color: textColor,
                      color: AppColors.black,
                      fontSize: AppStyle.fontSize15,
                    )),
          ),
        )
      ]),
    ),
  );

  ScaffoldMessenger.of(context).showSnackBar(SnackBar(
    content: _widget,
    backgroundColor: Colors.transparent,
    elevation: 0,
    duration: Duration(seconds: duration),
  ));
}

Future<bool> confirmAlertDialog(
  BuildContext context,
  String title,
  String message,
  String yesStr,
  String noStr, {
  bool barrierDismissible = false,
  bool isOptionsButton = false,
}) async {
  bool action = false;
  await showDialog<void>(
    context: context,
    barrierDismissible: barrierDismissible,
    builder: (context) {
      return Dialog(
        backgroundColor: Colors.white,
        shape: AppStyle().shapeRound,
        elevation: 4,
        child: Padding(
          padding: const EdgeInsets.symmetric(
            horizontal: 20,
            vertical: 15,
          ),
          child: IntrinsicWidth(
            child: IntrinsicHeight(
              child: Column(
                crossAxisAlignment: CrossAxisAlignment.start,
                children: <Widget>[
                  const Padding(
                    padding: EdgeInsets.only(top: 5),
                  ),
                  if (title != "")
                    Text(
                      title,
                      style: AppStyle().regularStyle.copyWith(
                            fontSize: AppStyle.fontSize16,
                            fontWeight: FontWeight.w600,
                            color: AppColors.black,
                          ),
                    ),
                  if (title != "")
                    const Padding(
                      padding: EdgeInsets.only(top: 5),
                    ),
                  Text(
                    message,
                    style: AppStyle().regularStyle.copyWith(
                          fontSize: AppStyle.fontSize14,
                          fontWeight: FontWeight.w400,
                          color: AppColors.black,
                        ),
                  ),
                  Padding(
                    padding: EdgeInsets.only(top: isOptionsButton ? 12 : 5),
                  ),
                  Row(
                    mainAxisAlignment:
                        (noStr.isEmpty || yesStr.isEmpty) && isOptionsButton
                            ? MainAxisAlignment.center
                            : MainAxisAlignment.end,
                    crossAxisAlignment: CrossAxisAlignment.end,
                    children: <Widget>[
                      if (noStr.isNotEmpty)
                        isOptionsButton
                            ? GetBuilder<ThemeController>(
                                builder: (themeController) {
                                return CustomElevatedButton(
                                  onPressed: () {
                                    action = false;
                                    Navigator.of(context, rootNavigator: true)
                                        .pop();
                                  },
                                  labelText: noStr,
                                  isWidthInfinity: false,
                                  labelTextStyle:
                                      AppStyle().textStyleLabel.copyWith(
                                            color: themeController
                                                        .getEnterpriseDetailsFromDB !=
                                                    null
                                                ? themeController
                                                                .getEnterpriseDetailsFromDB[
                                                            'primaryColor'] !=
                                                        null
                                                    ? Color(int.parse(
                                                        'FF' +
                                                            themeController
                                                                    .getEnterpriseDetailsFromDB[
                                                                'primaryColor'],
                                                        radix: 16))
                                                    : null
                                                : null,
                                            fontWeight: FontWeight.w600,
                                            fontSize: AppStyle.fontSize14,
                                          ),
                                  style: ElevatedButton.styleFrom(
                                    minimumSize: Size.zero,
                                    padding: const EdgeInsets.symmetric(
                                      horizontal: 32.0,
                                      vertical: 12.0,
                                    ),
                                    tapTargetSize:
                                        MaterialTapTargetSize.shrinkWrap,
                                    elevation: 0,
                                    shape: RoundedRectangleBorder(
                                      borderRadius:
                                          BorderRadius.circular(6.0),
                                    ),
                                  ).copyWith(
                                      backgroundColor: MaterialStateProperty
                                          .all(themeController
                                                      .getEnterpriseDetailsFromDB !=
                                                  null
                                              ? themeController
                                                              .getEnterpriseDetailsFromDB[
                                                          'secondaryColor'] !=
                                                      null
                                                  ? Color(int.parse(
                                                      'FF' +
                                                          themeController
                                                                  .getEnterpriseDetailsFromDB[
                                                              'secondaryColor'],
                                                      radix: 16))
                                                  : null
                                              : null)),
                                );
                              })
                            : TextButton(
                                child: Text(
                                  noStr,
                                  style: AppStyle().regularStyle.copyWith(
                                        fontSize: AppStyle.fontSize14,
                                        fontWeight: FontWeight.w400,
                                        color: AppColors.black,
                                      ),
                                ),
                                onPressed: () {
                                  action = false;
                                  Navigator.of(context, rootNavigator: true)
                                      .pop();
                                },
                              ),
                      Visibility(
                          visible: isOptionsButton &&
                              noStr.isNotEmpty &&
                              yesStr.isNotEmpty,
                          child: const SizedBox(
                            width: 16,
                          )),
                      if (yesStr.isNotEmpty)
                        isOptionsButton
                            ? CustomElevatedButton(
                                onPressed: () {
                                  action = true;
                                  Navigator.of(context, rootNavigator: true)
                                      .pop();
                                },
                                labelText: yesStr,
                                isWidthInfinity: false)
                            : TextButton(
                                child: GetBuilder<ThemeController>(
                                    builder: (themeLogic) {
                                  return Text(
                                    yesStr,
                                    style: AppStyle().regularStyle.copyWith(
                                        fontSize: AppStyle.fontSize14,
                                        fontWeight: FontWeight.w400,
                                        color: themeLogic
                                            .getSelectedPrimaryThemeColor),
                                  );
                                }),
                                onPressed: () {
                                  action = true;
                                  Navigator.of(context, rootNavigator: true)
                                      .pop();
                                },
                              ),
                    ],
                  )
                ],
              ),
            ),
          ),
        ),
      );
    },
  );
  return action;
}

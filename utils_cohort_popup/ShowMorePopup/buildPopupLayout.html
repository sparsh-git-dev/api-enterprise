LayoutBuilder buildPopupLayout(Offset offset) {
  return LayoutBuilder(builder: (context, constraints) {
    return GestureDetector(
      behavior: HitTestBehavior.translucent,
      onTap: () {
        dismiss();
      },
      child: Material(
        color: Colors.transparent,
        child: Stack(
          children: <Widget>[
            // triangle arrow
            Positioned(
              left: _showRect.left + _showRect.width / 2.0 - 7.5,
              top: _isDownArrow
                  ? offset.dy + _popupHeight
                  : offset.dy - arrowHeight,
              child: CustomPaint(
                size: Size(15.0, arrowHeight),
                painter: TrianglePainter(
                    isDownArrow: _isDownArrow, color: _backgroundColor),
              ),
            ),
            // popup content
            Positioned(
              left: offset.dx,
              top: offset.dy,
              child: Container(
                  padding: EdgeInsets.all(10),
                  decoration: BoxDecoration(
                      color: _backgroundColor,
                      borderRadius: _borderRadius,
                      boxShadow: [
                        BoxShadow(
                          color: Color(0xFF808080),
                          blurRadius: 1.0,
                        ),
                      ]),
                  child: Column(
                    crossAxisAlignment: CrossAxisAlignment.start,
                    children: [
                      Text(
                        _text,
                        style: _textStyle,
                      ),
                      SizedBox(height: _subTitle.length>1?70:40,
                        width: Get.width*0.45,
                        child: ListView.builder(
                          physics: NeverScrollableScrollPhysics(),
                            itemCount: _subTitle.length,
                            itemBuilder:(context,index){
                          return
                          Padding(
                            padding: const EdgeInsets.only(bottom: 10),
                            child: Row(
                              children: [
                                Container(
                                  height: 14,
                                  width: 14,
                                  decoration: BoxDecoration(
                                    border: Border.all(color:_color[index],
                                        width: 2
                                    ),
                                    color: _color[index].withOpacity(0.2),
                                  ),
                                ),
                                SizedBox(width: 5,),
                                Text(
                                  '${_subTitle[index]} : $_value',
                                  style: _textStyle,
                                ),
                              ],
                            ),
                          );
                        }),
                      ),
                    ],
                  )),
            )
          ],
        ),
      ),
    );
  });
}

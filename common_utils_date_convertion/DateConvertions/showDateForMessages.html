String showDateForMessages(String inputString, String inputFormat,
    {sendToday = false,sendYesterday = false}) {
  // workDuration(inputString, DateTime.now().toString());
  DateTime workedFromDT = DateFormat(inputFormat).parse(inputString, true);
  int days = DateTime.now().difference(workedFromDT).inDays;
  if (days <= 1) {
    final now = DateTime.now();
    final yesterday = DateTime(now.year, now.month, now.day - 1);
    final dayBeforeYesterday = DateTime(now.year, now.month, now.day - 2);
    if (workedFromDT.day == yesterday.day && sendYesterday) {
      return "Yesterday";
    }
    if (workedFromDT.day == yesterday.day && sendToday) {
      return "Yesterday";
    }
    if (workedFromDT.day == dayBeforeYesterday.day) {
      return dateParseLocalTimezone(
          inputString, "yyyy-MM-dd HH:mm:ss", "MMMM dd, yyyy");
    }
    if (sendToday) {
      return "Today";
    }
    return convertToAgo(workedFromDT);
  } else if (false) {
    ///making this false for now to match web cases
    return dateParseLocalTimezone(
        inputString, "yyyy-MM-dd HH:mm:ss", "E, HH:MM");
  } else {
    return dateParseLocalTimezone(
        inputString, "yyyy-MM-dd HH:mm:ss", "MMMM dd, yyyy");
  }
}

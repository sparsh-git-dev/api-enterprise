Widget matchingCandidateGetStatusChip(String status, {Color? chipColor}) {
  return chipColor != null
      ? Padding(
          padding: const EdgeInsets.only(bottom: 12, top: 0),
          child: Text(
            statusChip(status),
            style: AppStyle().regularStyle.copyWith(
                fontSize: AppStyle.fontSize14,
                fontWeight: FontWeight.w500,
                color:
                    (status == 'null' || status == '' || status == 'Shortlist')
                        ? AppColors.pending
                        : darkenColor(getOfferStatusTextColor(status), 20)),
          ),
        )
      : Container(
          height: 31,
          decoration: BoxDecoration(
            color: status == 'null'
                ? AppColors.applied.withOpacity(0.2)
                : getOfferStatusTextBackgroundColor(status,
                    chipColor: chipColor),
            borderRadius: BorderRadius.circular(7),
          ),
          child: RawChip(
            backgroundColor: status == 'null'
                ? AppColors.applied.withOpacity(0.2)
                : getOfferStatusTextBackgroundColor(status,
                    chipColor: chipColor),
            label: Padding(
              padding: const EdgeInsets.only(bottom: 2),
              child: Text(
                statusChip(status),
                style: AppStyle().regularStyle.copyWith(
                    fontSize: AppStyle.fontSize14,
                    fontWeight: FontWeight.w500,
                    color: (status == 'null' ||
                            status == '' ||
                            status == 'Shortlist')
                        ? AppColors.black
                        : darkenColor(getOfferStatusTextColor(status), 20)),
              ),
            ),
          ),
        );
}

showAssessmentDialog(BuildContext context, List questionList, List answerList) {
  return showDialog(
      context: context,
      barrierDismissible: false,
      builder: (context) {
        return Dialog(
          shape:
              RoundedRectangleBorder(borderRadius: BorderRadius.circular(12)),
          insetPadding: const EdgeInsets.all(14),
          child: SingleChildScrollView(
            child: Padding(
              padding: const EdgeInsets.symmetric(horizontal: 16, vertical: 16),
              child: Column(
                children: [
                  Row(
                    mainAxisSize: MainAxisSize.max,
                    mainAxisAlignment: MainAxisAlignment.spaceBetween,
                    children: [
                      Text(
                        EnterpriseStrings.profile_l067,
                        style: AppStyle().regularStyle.copyWith(
                              fontSize: AppStyle.fontSize18,
                              fontWeight: FontWeight.w800,
                            ),
                      ),
                      GestureDetector(
                          onTap: () {
                            Navigator.of(context).pop();
                          },
                          child: const Icon(
                            Icons.clear,
                            color: AppColors.subTitleColor,
                          ))
                    ],
                  ),
                  const SizedBox(height: 24),
                  ListView.builder(
                      shrinkWrap: true,
                      primary: false,
                      itemCount: questionList.length,
                      itemBuilder: ((context, index) {
                        return Column(
                          children: [
                            Align(
                              alignment: Alignment.centerLeft,
                              child: Text(
                                EnterpriseStrings.assessment_l002 +
                                    '${index + 1}' +
                                    '. ' +
                                    questionList[index],
                                style: AppStyle().regularStyle,
                              ),
                            ),
                            const SizedBox(height: 8),
                            Align(
                              alignment: Alignment.centerLeft,
                              child: GetBuilder<ThemeController>(
                                  builder: (themeLogic) {
                                return CustomReadMoreText(
                                  answerList[index].isNotEmpty
                                      ? EnterpriseStrings.assessment_l007 +
                                          answerList[index]
                                      : EnterpriseStrings.assessment_l007 + "-",
                                  colorClickableText: AppColors.midBlue,
                                  moreStyle: AppStyle()
                                      .viewMoreLessStyle
                                      .copyWith(
                                          color: themeLogic
                                              .getSelectedPrimaryThemeColor),
                                  lessStyle: AppStyle()
                                      .viewMoreLessStyle
                                      .copyWith(
                                          color: themeLogic
                                              .getSelectedPrimaryThemeColor),
                                  trimMode: TrimMode.Line,
                                  trimCollapsedText: EnterpriseStrings.b033,
                                  style: AppStyle().regularStyle.copyWith(
                                        fontSize: AppStyle.fontSize14,
                                        fontWeight: FontWeight.w400,
                                        color: AppColors.subTitleColor,
                                        height: 1.5,
                                      ),
                                  trimLines: 3,
                                  trimExpandedText: EnterpriseStrings.b034,
                                );
                              }),
                            ),
                            /*  Align(
                              alignment: Alignment.centerLeft,
                              child:

                              RichText(

                                  text: EnterpriseStrings.assessment_l007,

                                TextSpan(
                                  text: answerList[index].isNotEmpty

                                      : '-',
                                  style: AppStyle().regularStyle,
                                ),
                              ])),
                              */

                            const SizedBox(height: 24),
                          ],
                        );
                      }))

Future<List<TalentPoolData>?> fetchTalentPoolAsPerCandidate(
    candidateId, BuildContext context,
    {firstLoad = false, showLoader = false}) async {
  String url = ApiUrlConstants.TALENT_POOL_CANDIDATE_ADD_TO_POOLS +
      "/" +
      candidateId +
      "/" +
      pageNumber.toString() +
      '/10';
  final GlobalKey<State> _keyLoader = GlobalKey<State>();
  final usersResponse;
  if (pageNumber > 0 || showLoader) {
    usersResponse =
        await ApiServices().apiGetCallerWithLoader(url, context, _keyLoader);
  } else {
    var response;
    response = await ApiServices().apiGetCallerWithoutLoaderFullResponse(
      url,
      context,
    );
    if (response.statusCode == 200) {
      var responseJson = json.decode(response.body);
      totalCount = 0;
      totalCount = responseJson['totalCount'];
      update();
    }
    usersResponse = ApiServices().returnResponse(response, context);
  }
  setTalentPoolListingResponse(
      talentPoolDataFromMap(usersResponse), firstLoad);
  setIsLoading(false);
  setInitialLoad(false);
  return null;
}

Map<String, dynamic> toJson() {
  final Map<String, dynamic> data = new Map<String, dynamic>();
  data['benefits'] = this.benefits;
  data['boundary'] = this.boundary;
  data['budgetMaximum'] = this.budgetMaximum;
  data['budgetMinimum'] = this.budgetMinimum;
  data['city'] = this.city;
  data['country'] = this.country;
  data['currency'] = this.currency;
  data['customWeightage'] = this.customWeightage;
  data['id'] = this.id;
  data['jobCategory'] = this.jobCategory;
  data['jobDescriptionText'] = this.jobDescriptionText;
  data['jobDescriptionUrl'] = this.jobDescriptionUrl;
  data["jobEducationalQualificationsDtoList"] =
      jobEducationalQualificationsDtoList == null
          ? []
          : List<dynamic>.from(
              jobEducationalQualificationsDtoList!.map((x) => x.toMap()));

  data['jobExperienceLevel'] = this.jobExperienceLevel;
  data['jobTitle'] = this.jobTitle;
  data['jobType'] = this.jobType;
  data['workflowUniqueIdentifier'] = this.workflowUniqueIdentifier;
  data['limitForApplications'] = this.limitForApplications;
  data['limitForShortlists'] = this.limitForShortlists;
  data['maximumBudgetPerPaymentTerm'] = this.maximumBudgetPerPaymentTerm;
  data['minimumBudgetPerPaymentTerm'] = this.minimumBudgetPerPaymentTerm;
  data['minimumEducation'] = this.minimumEducation;
  data['mspEnterpriseId'] = this.mspEnterpriseId;
  data['neuroClassification'] = this.neuroClassification;
  data['numberOfOpenings'] = this.numberOfOpenings;
  data['overallExperienceMaximum'] = this.overallExperienceMaximum;
  data['overallExperienceMinimum'] = this.overallExperienceMinimum;
  data['paymentTerm'] = this.paymentTerm;
  data['priorityPoolBoundary'] = this.priorityPoolBoundary;
  data['questions'] = this.questions;
  data['shift'] = this.shift;
  data['socialMediaUrl'] = this.socialMediaUrl;
  data['state'] = this.state;
  data['status'] = this.status;
  data['validFrom'] = this.validFrom;
  data['validTill'] = this.validTill;
  data['vendorNeutrality'] = this.vendorNeutrality;
  data['allowedToWorkInOtherJobs'] = this.allowedToWorkInOtherJobs;
  data['workFrom'] = this.workFrom;
  return data;
}

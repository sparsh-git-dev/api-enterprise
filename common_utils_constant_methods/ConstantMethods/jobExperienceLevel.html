String jobExperienceLevel(jobExpLevel) {
  return jobExpLevel == 'IN'
      ? 'Internship'
      : jobExpLevel == 'EL'
          ? 'Entry level'
          : jobExpLevel == 'AT'
              ? 'Associate'
              : jobExpLevel == 'ML'
                  ? 'Mid-Senior Level'
                  : jobExpLevel == 'DR'
                      ? 'Director'
                      : jobExpLevel == 'ET'
                          ? 'Executive'
                          : '-';
}

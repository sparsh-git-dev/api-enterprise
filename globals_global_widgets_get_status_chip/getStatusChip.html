Widget getStatusChip(
  String offerStatus, {
  fromTalentPoolChanges = false,
  fromMatchingCandidate = false,
  bool isAvailableStatus = false,
  blackText = false,
  colorOpacity = 0.2,
  required BuildContext context,
  chipLabelText = '',
  String candName = "",
}) {
  StatusColorController _statusController = Get.put(StatusColorController());
  return Container(
    height: Responsive.isTablet(context) ? 43 : 30,
    decoration: BoxDecoration(
      color: isAvailableStatus
          ? getStatusBGColor(offerStatus, _statusController)
          : chipLabelText != ''
              ? getCustomTextColor(chipLabelText)
              : fromTalentPoolChanges
                  ? getTalentPoolStatusTextColor(offerStatus)
                  // : (offerStatus == 'MCO' || offerStatus == 'MCA')
                  //     ? getMyCandidatesStatusTextColor(offerStatus)
                  //         .withOpacity(colorOpacity)
                  : fromMatchingCandidate
                      ? getMyCandidatesStatusTextColor(offerStatus)
                      : getOfferStatusTextColor(offerStatus)
                          .withOpacity(colorOpacity),
      borderRadius: BorderRadius.circular(20),
    ),
    child: RawChip(
      backgroundColor: isAvailableStatus
          ? getStatusBGColor(offerStatus, _statusController)
          : chipLabelText != ''
              ? getCustomTextColor(chipLabelText)
              : fromTalentPoolChanges
                  ? getTalentPoolStatusTextColor(offerStatus)
                  : (offerStatus == 'MCO' || offerStatus == 'MCA')
                      ? getMyCandidatesStatusTextColor(offerStatus)
                      // .withOpacity(colorOpacity)
                      : getOfferStatusTextColor(offerStatus)
                          .withOpacity(colorOpacity),
      label: Padding(
        padding: const EdgeInsets.only(bottom: 2),
        child: Text(
          isAvailableStatus
              ? getStatus(offerStatus, _statusController)
              : chipLabelText != ''
                  ? chipLabelText
                  : fromTalentPoolChanges
                      ? statusTalentPoolChangesChip(offerStatus)
                      : statusChip(offerStatus),
          style: AppStyle().regularStyle.copyWith(
              fontSize: AppStyle.fontSize14,
              fontWeight: FontWeight.w500,
              color: isAvailableStatus
                  ? getStatusTextColor(offerStatus, _statusController)
                  : fromTalentPoolChanges || blackText
                      ? AppColors.black
                      : offerStatus == 'P' || offerStatus == 's'
                          ? AppColors.pending2
                          : (offerStatus == 'MCO' || offerStatus == 'MCA')
                              ? getMatchingCandidatesStatusTextColor(
                                  offerStatus)
                              : getOfferStatusTextColor(offerStatus)),
        ),
      ),
    ),
  );
}
